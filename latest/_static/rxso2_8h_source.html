58 /// RxSo2 base type - implements RxSo2 class but is storage agnostic
59 ///
60 /// This class implements the group ``R+ x SO(2)``, the direct product of the
61 /// group of positive scalar 2x2 matrices (= isomorph to the positive
62 /// real numbers) and the two-dimensional special orthogonal group SO(2).
63 /// Geometrically, it is the group of rotation and scaling in two dimensions.
64 /// As a matrix groups, R+ x SO(2) consists of matrices of the form ``s * R``
65 /// where ``R`` is an orthogonal matrix with ``det(R) = 1`` and ``s > 0``
66 /// being a positive real number. In particular, it has the following form:
67 ///
68 /// | s * cos(theta) s * -sin(theta) |
69 /// | s * sin(theta) s * cos(theta) |
70 ///
71 /// where ``theta`` being the rotation angle. Internally, it is represented by
72 /// the first column of the rotation matrix, or in other words by a non-zero
73 /// complex number.
74 ///
75 /// R+ x SO(2) is not compact, but a commutative group. First it is not compact
76 /// since the scale factor is not bound. Second it is commutative since
77 /// ``sR(alpha, s1) * sR(beta, s2) = sR(beta, s2) * sR(alpha, s1)``, simply
78 /// because ``alpha + beta = beta + alpha`` and ``s1 * s2 = s2 * s1`` with
79 /// ``alpha`` and ``beta`` being rotation angles and ``s1``, ``s2`` being scale
80 /// factors.
81 ///
82 /// This class has the explicit class invariant that the scale ``s`` is not
83 /// too close to either zero or infinity. Strictly speaking, it must hold that:
84 ///
85 /// ``complex().norm() >= Constants::epsilon()`` and
86 /// ``1. / complex().norm() >= Constants::epsilon()``.
87 ///
88 /// In order to obey this condition, group multiplication is implemented with
89 /// saturation such that a product always has a scale which is equal or greater
90 /// this threshold.
91 template <class TDerived>
92 class RxSo2Base {
93  public:
94  static int constexpr kOptions = Eigen::internal::traits<TDerived>::kOptions;

96  using ComplexType = typename Eigen::internal::traits<TDerived>::ComplexType;
97  using ComplexTemporaryType = Eigen::Matrix<Scalar, 2, 1, kOptions>;

174  /// Returns group inverse.
175  ///
176  SOPHUS_FUNC [[nodiscard]] RxSo2<Scalar> inverse() const {
177  Scalar squared_scale = complex().squaredNorm();
178  Eigen::Vector2<Scalar> xy = complex() / squared_scale;
179  return RxSo2<Scalar>(xy.x(), -xy.y());

182  /// Logarithmic map
183  ///
184  /// Computes the logarithm, the inverse of the group exponential which maps
185  /// element of the group (scaled rotation matrices) to elements of the tangent
186  /// space (rotation-vector plus logarithm of scale factor).
187  ///
188  /// To be specific, this function computes ``vee(logmat(.))`` with
189  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
190  /// of RxSo2.
191  ///
192  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
193  using std::log;
194  Tangent theta_sigma;
195  theta_sigma[1] = log(scale());
196  theta_sigma[0] = So2<Scalar>(complex()).log();
197  return theta_sigma;
198  }

200  /// Returns 2x2 matrix representation of the instance.
201  ///
202  /// For RxSo2, the matrix representation is an scaled orthogonal matrix ``sR``
203  /// with ``det(R)=s^2``, thus a scaled rotation matrix ``R`` with scale
204  /// ``s``.
205  ///
206  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
207  Transformation s_r;
208  // clang-format off
209  s_r << complex()[0], -complex()[1],
210  complex()[1], complex()[0];
211  // clang-format on
212  return s_r;
213  }

224  /// Group multiplication, which is rotation concatenation and scale
225  /// multiplication.
226  ///
227  /// Note: This function performs saturation for products close to zero in
228  /// order to ensure the class invariant.
229  ///
230  template <class TOtherDerived>
231  SOPHUS_FUNC RxSo2Product<TOtherDerived> operator*(
232  RxSo2Base<TOtherDerived> const& other) const {
233  using ResultT = ReturnScalar<TOtherDerived>;
234 
235  Scalar lhs_real = complex().x();
236  Scalar lhs_imag = complex().y();
237  typename TOtherDerived::Scalar const& rhs_real = other.complex().x();
238  typename TOtherDerived::Scalar const& rhs_imag = other.complex().y();
239  /// complex multiplication
240  typename RxSo2Product<TOtherDerived>::ComplexType result_complex(
241  lhs_real * rhs_real - lhs_imag * rhs_imag,
242  lhs_real * rhs_imag + lhs_imag * rhs_real);
243 
244  const ResultT squared_scale = result_complex.squaredNorm();
245 
246  if (squared_scale < kEpsilon<ResultT> * kEpsilon<ResultT>) {
247  /// Saturation to ensure class invariant.
248  result_complex.normalize();
249  result_complex *= kEpsilonPlus<ResultT>;
250  }
251  if (squared_scale > Scalar(1.) / (kEpsilon<ResultT> * kEpsilon<ResultT>)) {
252  /// Saturation to ensure class invariant.
253  result_complex.normalize();
254  result_complex /= kEpsilonPlus<ResultT>;
255  }
256  return RxSo2Product<TOtherDerived>(result_complex);
257  }

259  /// Group action on 2-points.
260  ///
261  /// This function rotates a 2 dimensional point ``p`` by the So2 element
262  /// ``bar_R_foo`` (= rotation matrix) and scales it by the scale factor ``s``:
263  ///
264  /// ``p_bar = s * (bar_R_foo * p_foo)``.
265  ///
266  template <
267  typename TPointDerived,
268  typename = typename std::enable_if<
269  IsFixedSizeVector<TPointDerived, 2>::value>::type>
270  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
271  Eigen::MatrixBase<TPointDerived> const& p) const {
272  return matrix() * p;
273  }

287  /// Group action on lines.
288  ///
289  /// This function rotates a parameterized line ``l(t) = o + t * d`` by the So2
290  /// element and scales it by the scale factor
291  ///
292  /// Origin ``o`` is rotated and scaled
293  /// Direction ``d`` is rotated (preserving it's norm)
294  ///
295  SOPHUS_FUNC Line operator*(Line const& l) const {
296  return Line((*this) * l.origin(), (*this) * l.direction() / scale());
297  }

299  /// Group action on hyper-planes.
300  ///
301  /// This function rotates a hyper-plane ``n.x + d = 0`` by the So2
302  /// element and scales offset by the scale factor
303  ///
304  /// Normal vector ``n`` is rotated
305  /// Offset ``d`` is scaled
306  ///
307  /// Note that in 2d-case hyper-planes are just another parametrization of
308  /// lines
309  ///
310  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
311  auto const this_scale = scale();
312  return Hyperplane(
313  (*this) * p.normal() / this_scale, this_scale * p.offset());
314  }

554  /// Returns derivative of exp(x) wrt. ``x``

601  /// Group exponential
602  ///
603  /// This functions takes in an element of tangent space (= rotation angle
604  /// plus logarithm of scale) and returns the corresponding element of the
605  /// group RxSo2.
606  ///
607  /// To be more specific, this function computes ``expmat(hat(theta))``
608  /// with ``expmat(.)`` being the matrix exponential and ``hat(.)`` being the
609  /// hat()-operator of RSO2.
610  ///
611  SOPHUS_FUNC static RxSo2<Scalar> exp(Tangent const& vec_a) {
612  using std::exp;
613  using std::max;
614  using std::min;
615 
616  Scalar const theta = vec_a[0];
617  Scalar const sigma = vec_a[1];
618  Scalar s = exp(sigma);
619  // Ensuring proper scale
620  s = max(s, kEpsilonPlus<Scalar>);
621  s = min(s, Scalar(1.) / kEpsilonPlus<Scalar>);
622  Eigen::Vector2<Scalar> z = So2<Scalar>::exp(theta).unitComplex();
623  z *= s;
624  return RxSo2<Scalar>(z);
625  }

627  /// Returns the ith infinitesimal generators of ``R+ x SO(2)``.
628  ///
629  /// The infinitesimal generators of RxSo2 are:
630  ///
631  /// ```
632  /// | 0 -1 |
633  /// G_0 = | 1 0 |
634  ///
635  /// | 1 0 |
636  /// G_1 = | 0 1 |
637  /// ```
638  ///
639  /// Precondition: ``i`` must be 0, or 1.
640  ///
641  SOPHUS_FUNC static Transformation generator(int i) {
642  SOPHUS_ASSERT(i >= 0 && i <= 1, "i should be 0 or 1.");
643  Tangent e;
644  e.setZero();
645  e[i] = Scalar(1);
646  return hat(e);
647  }
648 
649  /// hat-operator
650  ///
651  /// It takes in the 2-vector representation ``a`` (= rotation angle plus
652  /// logarithm of scale) and returns the corresponding matrix representation
653  /// of Lie algebra element.
654  ///
655  /// Formally, the hat()-operator of RxSo2 is defined as
656  ///
657  /// ``hat(.): R^2 -> R^{2x2}, hat(a) = sum_i a_i * G_i`` (for i=0,1,2)
658  ///
659  /// with ``G_i`` being the ith infinitesimal generator of RxSo2.
660  ///
661  /// The corresponding inverse is the vee()-operator, see below.
662  ///
663  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
664  Transformation mat_a;
665  // clang-format off
666  mat_a << vec_a(1), -vec_a(0),
667  vec_a(0), vec_a(1);
668  // clang-format on
669  return mat_a;
670  }
671 
672  /// Lie bracket
673  ///
674  /// It computes the Lie bracket of RxSO(2). To be more specific, it computes
675  ///
676  /// ``[omega_1, omega_2]_rxso2 := vee([hat(omega_1), hat(omega_2)])``
677  ///
678  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
679  /// hat()-operator and ``vee(.)`` the vee()-operator of RxSo2.
680  ///
681  SOPHUS_FUNC static Tangent lieBracket(
682  Tangent const& /*unused*/, Tangent const& /*unused*/) {
683  Eigen::Vector2<Scalar> res;
684  res.setZero();
685  return res;
686  }
687 
688  /// Draw uniform sample from RxSO(2) manifold.
689  ///
690  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
691  /// hence the scale is in [0.5, 2)].
692  ///
693  template <class TUniformRandomBitGenerator>
694  static RxSo2 sampleUniform(TUniformRandomBitGenerator& generator) {
695  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
696  using std::exp2;
697  return RxSo2(
698  exp2(uniform(generator)), So2<Scalar>::sampleUniform(generator));
699  }
700 
701  /// vee-operator
702  ///
703  /// It takes the 2x2-matrix representation ``Omega`` and maps it to the
704  /// corresponding vector representation of Lie algebra.
705  ///
706  /// This is the inverse of the hat()-operator, see above.
707  ///
708  /// Precondition: ``Omega`` must have the following structure:
709  ///
710  /// | d -x |
711  /// | x d |
712  ///
713  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
714  using std::abs;
715  return Tangent(omega(1, 0), omega(0, 0));
716  }
717 
718  protected:
719  SOPHUS_FUNC ComplexMember& mutComplex() { return complex_; }
720 
721  ComplexMember complex_; // NOLINT
722 };
