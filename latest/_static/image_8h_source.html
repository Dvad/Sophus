28 /// A image with write access to pixels and exclusive ownership. There is no
29 /// copy constr / copy assignment, but move constr / assignment.
30 ///
31 /// Content from a MutImage can be moved into an Image.
32 ///
33 /// Type is nullable. In that case `this->isEmpty()` is true.
34 ///
35 /// Similar to Pangolin::ManagedImage.
36 template <
37  class TPixel,
38  template <class> class TAllocator = Eigen::aligned_allocator>
39 class MutImage : public MutImageView<TPixel> {
40  public:
41  struct TypedDeleterImpl {
42  TypedDeleterImpl(size_t num_bytes) : num_bytes(num_bytes) {}
43 
44  void operator()(TPixel* p) const {
45  if (p != nullptr) {
46  TAllocator<TPixel>().deallocate(p, num_bytes / sizeof(TPixel));
47  }
48  }
49 
50  size_t num_bytes;
51  };
52 
53  /// Deleter for MutImage.
54  struct Deleter {
55  Deleter(TypedDeleterImpl image_deleter) : image_deleter(image_deleter) {}
56 
57  void operator()(uint8_t* p) const {
58  if (image_deleter) {
59  (*image_deleter)(reinterpret_cast<TPixel*>(p));
60  }
61  }
62 
63  std::optional<TypedDeleterImpl> image_deleter;
64  };
65 
66  template <class TT, template <class> class TAllocator2T>
67  friend class Image;

69  /// Constructs empty image.
70  MutImage() = default;
71 
72  /// Creates new image with given shape.
73  ///
74  /// If shape is not empty, memory allocation will happen.
75  explicit MutImage(ImageShape shape) : MutImageView<TPixel>(shape, nullptr) {
76  if (shape.sizeBytes() != 0u) {
77  SOPHUS_ASSERT_EQ(shape.sizeBytes() % sizeof(TPixel), 0);
78  this->shared_.reset(
79  (uint8_t*)(TAllocator<TPixel>().allocate(
80  shape.sizeBytes() / sizeof(TPixel))),
81  Deleter(TypedDeleterImpl(shape.sizeBytes())));
82  }
83  this->ptr_ = reinterpret_cast<TPixel*>(this->shared_.get());
84  }
85 
86  /// Creates new contiguous image with given size.
87  ///
88  /// If shape is not empty, memory allocation will happen.
89  explicit MutImage(sophus::ImageSize size)
90  : MutImage(ImageShape::makeFromSize<TPixel>(size)) {}
91 
92  /// Creates contiguous copy from view.
93  ///
94  /// If view is not empty, memory allocation will happen.
95  [[nodiscard]] static MutImage makeCopyFrom(ImageView<TPixel> const& view) {
96  MutImage image(view.imageSize());
97  image.copyDataFrom(view);
98  return image;
99  }

101  /// Creates new MutImage given view and unary transform function.
102  ///
103  /// mut_image(u, v) = unary_op(view(u, v));
104  template <class TOtherPixel, class TUnaryOperation>
105  static MutImage makeFromTransform(
106  ImageView<TOtherPixel> view, TUnaryOperation const& unary_op) {
107  MutImage mut_image(view.imageSize());
108  mut_image.transformFrom(view, unary_op);
109  return mut_image;
110  }

112  /// Creates new MutImage given two views and binary transform function.
113  ///
114  /// mut_image(u, v) = binary_op(lhs(u, v), rhs(u, v));
115  template <class TLhsPixel, class TRhsPixel, class TBinaryOperation>
116  static MutImage makeFromTransform(
117  ImageView<TLhsPixel> lhs,
118  ImageView<TRhsPixel> rhs,
119  TBinaryOperation const& binary_op) {
120  MutImage mut_image(lhs.imageSize());
121  mut_image.transformFrom(lhs, rhs, binary_op);
122  return mut_image;
123  }

125  // Begin(Rule of 5):
126  // https://en.cppreference.com/w/cpp/language/rule_of_three
127 
128  /// Destructor
129  ~MutImage() { reset(); }
130 
131  /// Not copy constructable
132  MutImage(MutImage<TPixel> const& other) = delete;
133 
134  /// Not copy assignable
135  MutImage& operator=(MutImage const&) = delete;
136 
137  /// Move constructor - is cheap - no memory allocations.
138  MutImage(MutImage&& img) noexcept
139  : MutImageView<TPixel>(img.viewMut()), shared_(std::move(img.shared_)) {
140  this->ptr_ = reinterpret_cast<TPixel*>(shared_.get());
141  img.setViewToEmpty();
142  }
143 
144  /// Move assignment
145  MutImage& operator=(MutImage&& img) noexcept {
146  reset();
147  this->shape_ = img.shape_;
148  this->shared_ = std::move(img.shared_);
149  this->ptr_ = reinterpret_cast<TPixel*>(shared_.get());
150  img.setViewToEmpty();
151  return *this;
152  }
153  // End (Rule of 5)
154 
155  [[nodiscard]] MutImageView<TPixel> viewMut() const {
156  return MutImageView<TPixel>(this->shape(), this->ptrMut());
157  }
158 
159  /// Swaps img and this.
160  void swap(MutImage& img) {
161  std::swap(img.shape_, this->shape_);
162  std::swap(img.ptr_, this->ptr_);
163  std::swap(img.shared_, this->shared_);
164  }
165 
166  /// Clears image.
167  ///
168  /// If image was not empty, memory deallocations will happen.
169  void reset() {
170  shared_.reset();
171  this->setViewToEmpty();
172  }
173 
174  protected:
175  /// Leaks memory and returns deleter.
176  Deleter leakAndReturnDeleter() {
177  SOPHUS_ASSERT(!this->isEmpty());
178  Deleter& del = *std::get_deleter<Deleter>(shared_);
179  Deleter del_copy = del;
180  del.image_deleter.reset();
181  return del_copy;
182  }
183 
184  /// MutImage has unique ownership, and hence behaves like a unique_ptr. As an
185  /// implementation detail, we use a shared_ptr here, so it will be easy to
186  /// support moving a Image with unique ownership at runtime into a
187  /// MutImage.
188  /// Class invariant: shared_.use_count() == 0 || shared_.use_count() == 1.
189  std::shared_ptr<uint8_t> shared_; // NOLINT
190 };

195 /// Image read-only access to pixels and shared ownership, hence cheap to copy.
196 /// Type is nullable.
197 ///
198 /// Image has close interop with RuntimeImage (see below).
199 template <
200  class TPixel,
201  template <class> class TAllocator = Eigen::aligned_allocator>
202 class Image : public ImageView<TPixel> {
203  public:
204  /// Constructs empty image.
205  Image() = default;
206 
207  /// Moves MutImage into this.
208  /// By design not "explicit".
209  Image(MutImage<TPixel, TAllocator>&& image) noexcept
210  : ImageView<TPixel>(image.view()) {
211  if (!image.isEmpty()) {
212  this->shared_ = std::move(image.shared_);
213  this->ptr_ = reinterpret_cast<TPixel*>(shared_.get());
214  image.setViewToEmpty();
215  }
216  }
217 
218  /// Creates contiguous copy from view.
219  ///
220  /// If view is not empty, memory allocation will happen.
221  [[nodiscard]] static Image makeCopyFrom(ImageView<TPixel> const& view) {
222  return Image(MutImage<TPixel>::makeCopyFrom(view));
223  }
224 
225  /// Allocated and generates image from provided function taking u,v indices
226  ///
227  /// Memory allocation will happen.
228  template <class TUVOperation>
229  [[nodiscard]] static Image makeGenerative(
230  ImageSize size, TUVOperation const& uv_op) {
231  MutImage<TPixel> mut_image(size);
232  mut_image.generate(uv_op);
233  return mut_image;
234  }

236  /// Creates new Image given view and unary transform function.
237  ///
238  /// image(u, v) = unary_op(view(u, v));
239  template <class TOtherPixel, class TUnaryOperation>
240  static Image makeFromTransform(

245  /// Creates new Image given two views and binary transform function.
246  ///
247  /// image(u, v) = binary_op(lhs(u, v), rhs(u, v));
248  template <class TLhsPixel, class TRhsPixel, class TBinaryOperation>
249  static Image makeFromTransform(
250  ImageView<TLhsPixel> lhs,
251  ImageView<TRhsPixel> rhs,
252  TBinaryOperation const& binary_op) {
253  return MutImage<TPixel>::makeFromTransform(lhs, rhs, binary_op);
254  }
255 
256  [[nodiscard]] size_t useCount() const { return shared_.use_count(); }
257 
258  /// Sets Image instance to empty. Reduced use count by one.
259  ///
260  /// If use count goes to zero, deallocation happens.
261  ///
262  /// No-op if empty.
263  void reset() {
264  shared_.reset();
265  this->setViewToEmpty();
266  }
267 
268  private:
269  template <class TT, template <class> class TAllocator2T>
270  friend class MutImage;
271 
272  template <class TPredicate, template <class> class TAllocator2T>
273  friend class RuntimeImage;
274 
275  explicit Image(ImageView<TPixel> view) : ImageView<TPixel>(view) {}
276 
277  Image(ImageView<TPixel> view, std::shared_ptr<uint8_t> const& shared)
278  : ImageView<TPixel>(view), shared_(shared) {}
279 
280  std::shared_ptr<uint8_t> shared_;
281 };
