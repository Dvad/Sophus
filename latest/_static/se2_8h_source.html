59 /// Se2 base type - implements Se2 class but is storage agnostic.
60 ///
61 /// SE(2) is the group of rotations and translation in 2d. It is the
62 /// semi-direct product of SO(2) and the 2d Euclidean vector space. The class
63 /// is represented using a composition of SO2Group for rotation and a 2-vector
64 /// for translation.
65 ///
66 /// SE(2) is neither compact, nor a commutative group.
67 ///
68 /// - 3x3 Eigen::Matrix representation:
69 ///
70 /// ```
71 /// | R t |
72 /// | o 1 |
73 /// ```
74 ///
75 /// where ``R`` is a 2x2 rotation matrix, ``t`` a translation 2-vector and
76 /// ``o`` a 2-column vector of zeros.
77 ///
78 /// - Tangent 3-vector: [upsilon, theta],
79 ///
80 /// where ``upsilon`` is the translational velocity 2-vector and ``theta`` the
81 /// rotational velocity.
82 ///
83 /// - Internal 4-vector representation: [t0, t1, zr, zi],
84 ///
85 /// with ``t0, t1`` are the translational components, and ``zr``
86 /// the real part and ``zi`` the imaginary part of a unit length complex number.
87 ///
88 /// See SO2Group for more details of the rotation representation in 2d.
89 ///
90 template <class TDerived>
91 class Se2Base {
92  public:
93  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
94  using TranslationType =
95  typename Eigen::internal::traits<TDerived>::TranslationType;
96  using So2Type = typename Eigen::internal::traits<TDerived>::So2Type;

99  /// (two for translation, three for rotation).

101  /// Number of internal parameters used (tuple for complex, two for

103  static int constexpr kNumParameters = 4;
104  /// Group transformations are 3x3 matrices.

106  /// Points are 2-dimensional
107  static int constexpr kPointDim = 2;

112  using Hyperplane = Eigen::Hyperplane<Scalar, 2>;

133  /// Adjoint transformation
134  ///
135  /// This function return the adjoint transformation ``Ad`` of the group
136  /// element ``A`` such that for all ``x`` it holds that
137  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
138  ///
139  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
140  Eigen::Matrix<Scalar, 2, 2> const& mat_r = so2().matrix();
141  Transformation res;
142  res.setIdentity();
143  res.template topLeftCorner<2, 2>() = mat_r;
144  res(0, 2) = translation()[1];
145  res(1, 2) = -translation()[0];
146  return res;
147  }

158  /// Returns derivative of this * exp(x) wrt x at x=0.
159  ///
160  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>
161  dxThisMulExpXAt0() const {
162  Eigen::Matrix<Scalar, kNumParameters, kDoF> dx;
163  Eigen::Vector2<Scalar> const z = unitComplex();
164  Scalar o(0);
165  // clang-format off
166  dx(0, 0) = o; dx(1, 0) = o; dx(2, 0) = z[0]; dx(3, 0) = z[1];
167  dx(0, 1) = o; dx(1, 1) = o; dx(2, 1) = -z[1]; dx(3, 1) = z[0];
168  dx(0, 2) = -z[1]; dx(1, 2) = z[0]; dx(2, 2) = o; dx(3, 2) = o;
169  // clang-format on
170  return dx;
171  }

173  /// Returns derivative of log(this^{-1} * x) by x at x=this.
174  ///
175  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>
176  dxLogThisInvTimesXAtThis() const {
177  Eigen::Matrix<Scalar, kDoF, kNumParameters> d;
178  d.template block<2, 2>(0, 0).setZero();
179  d.template block<2, 2>(0, 2) = so2().inverse().matrix();
180  d.template block<1, 2>(2, 0) = so2().dxLogThisInvTimesXAtThis();
181  d.template block<1, 2>(2, 2).setZero();
182  return d;
183  }

192  /// Logarithmic map
193  ///
194  /// Computes the logarithm, the inverse of the group exponential which maps
195  /// element of the group (rigid body transformations) to elements of the
196  /// tangent space (twist).
197  ///
198  /// To be specific, this function computes ``vee(logmat(.))`` with
199  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
200  /// of SE(2).
201  ///
202  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
203  using std::abs;
204 
205  Tangent vec_upsilon_theta;
206  Scalar theta = so2().log();
207  vec_upsilon_theta[2] = theta;
208  Scalar halftheta = Scalar(0.5) * theta;
209  Scalar halftheta_by_tan_of_halftheta;
210 
211  Eigen::Vector2<Scalar> z = so2().unitComplex();
212  Scalar real_minus_one = z.x() - Scalar(1.);
213  if (abs(real_minus_one) < kEpsilon<Scalar>) {
214  halftheta_by_tan_of_halftheta =
215  Scalar(1.) - Scalar(1. / 12) * theta * theta;
216  } else {
217  halftheta_by_tan_of_halftheta = -(halftheta * z.y()) / (real_minus_one);
218  }
219  Eigen::Matrix<Scalar, 2, 2> v_inv;
220  v_inv << halftheta_by_tan_of_halftheta, halftheta, -halftheta,
221  halftheta_by_tan_of_halftheta;
222  vec_upsilon_theta.template head<2>() = v_inv * translation();
223  return vec_upsilon_theta;
224  }

232  /// Returns 3x3 matrix representation of the instance.
233  ///
234  /// It has the following form:
235  ///
236  /// ```
237  /// | R t |
238  /// | o 1 |
239  /// ```
240  ///
241  /// where ``R`` is a 2x2 rotation matrix, ``t`` a translation 2-vector and
242  /// ``o`` a 2-column vector of zeros.
243  ///
244  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
245  Transformation homogenious_matrix;
246  homogenious_matrix.template topLeftCorner<2, 3>() = matrix2x3();
247  homogenious_matrix.row(2) =
248  Eigen::Matrix<Scalar, 1, 3>(Scalar(0), Scalar(0), Scalar(1));
249  return homogenious_matrix;
250  }

280  /// Group action on 2-points.
281  ///
282  /// This function rotates and translates a two dimensional point ``p`` by the
283  /// SE(2) element ``bar_transform_foo = (bar_R_foo, t_bar)`` (= rigid body
284  /// transformation):
285  ///
286  /// ``p_bar = bar_R_foo * p_foo + t_bar``.
287  ///
288  template <
289  typename TPointDerived,
290  typename = typename std::enable_if<
291  IsFixedSizeVector<TPointDerived, 2>::value>::type>
292  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
293  Eigen::MatrixBase<TPointDerived> const& p) const {
294  return so2() * p + translation();
295  }
296 
297  /// Group action on homogeneous 2-points. See above for more details.
298  ///
299  template <
300  typename THPointDerived,
301  typename = typename std::enable_if<
302  IsFixedSizeVector<THPointDerived, 3>::value>::type>
303  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
304  Eigen::MatrixBase<THPointDerived> const& p) const {
305  PointProduct<THPointDerived> const tp =
306  so2() * p.template head<2>() + p(2) * translation();
307  return HomogeneousPointProduct<THPointDerived>(tp(0), tp(1), p(2));
308  }

310  /// Group action on lines.
311  ///
312  /// This function rotates and translates a parametrized line
313  /// ``l(t) = o + t * d`` by the SE(2) element:
314  ///
315  /// Origin ``o`` is rotated and translated using SE(2) action
316  /// Direction ``d`` is rotated using SO(2) action
317  ///
318  SOPHUS_FUNC Line operator*(Line const& l) const {
319  return Line((*this) * l.origin(), so2() * l.direction());
320  }
321 
322  /// Group action on hyper-planes.
323  ///
324  /// This function rotates a hyper-plane ``n.x + d = 0`` by the Se2
325  /// element:
326  ///
327  /// Normal vector ``n`` is rotated
328  /// Offset ``d`` is adjusted for translation
329  ///
330  /// Note that in 2d-case hyper-planes are just another parametrization of
331  /// lines
332  ///
333  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
334  Hyperplane const rotated = so2() * p;
335  return Hyperplane(
336  rotated.normal(),
337  rotated.offset() - translation().dot(rotated.normal()));
338  }

552  /// Returns derivative of exp(x) wrt. x.
553  ///
554  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
555  Tangent const& upsilon_theta) {
556  using std::abs;
557  using std::cos;
558  using std::pow;
559  using std::sin;
560  Eigen::Matrix<Scalar, kNumParameters, kDoF> dx;
561  Eigen::Vector<Scalar, 2> upsilon = upsilon_theta.template head<2>();
562  Scalar theta = upsilon_theta[2];
563 
564  if (abs(theta) < kEpsilon<Scalar>) {
565  Scalar const o(0);
566  Scalar const i(1);
567 
568  // clang-format off
569  dx << o, o, o,
570  o, o, i,
571  i, o, -Scalar(0.5) * upsilon[1],
572  o, i, Scalar(0.5) * upsilon[0];
573  // clang-format on
574  return dx;
575  }
576 
577  Scalar const c0 = sin(theta);
578  Scalar const c1 = cos(theta);
579  Scalar const c2 = 1.0 / theta;
580  Scalar const c3 = c0 * c2;
581  Scalar const c4 = -c1 + Scalar(1);
582  Scalar const c5 = c2 * c4;
583  Scalar const c6 = c1 * c2;
584  Scalar const c7 = pow(theta, -2);
585  Scalar const c8 = c0 * c7;
586  Scalar const c9 = c4 * c7;
587 
588  Scalar const o = Scalar(0);
589  dx(0, 0) = o;
590  dx(0, 1) = o;
591  dx(0, 2) = -c0;
592 
593  dx(1, 0) = o;
594  dx(1, 1) = o;
595  dx(1, 2) = c1;
596 
597  dx(2, 0) = c3;
598  dx(2, 1) = -c5;
599  dx(2, 2) =
600  -c3 * upsilon[1] + c6 * upsilon[0] - c8 * upsilon[0] + c9 * upsilon[1];
601 
602  dx(3, 0) = c5;
603  dx(3, 1) = c3;
604  dx(3, 2) =
605  c3 * upsilon[0] + c6 * upsilon[1] - c8 * upsilon[1] - c9 * upsilon[0];
606  return dx;
607  }

642  /// Group exponential
643  ///
644  /// This functions takes in an element of tangent space (= twist ``a``) and
645  /// returns the corresponding element of the group SE(2).
646  ///
647  /// The first two components of ``a`` represent the translational part
648  /// ``upsilon`` in the tangent space of SE(2), while the last three components
649  /// of ``a`` represents the rotation vector ``omega``.
650  /// To be more specific, this function computes ``expmat(hat(a))`` with
651  /// ``expmat(.)`` being the matrix exponential and ``hat(.)`` the hat-operator
652  /// of SE(2), see below.
653  ///
654  SOPHUS_FUNC static Se2<Scalar> exp(Tangent const& vec_a) {
655  Scalar theta = vec_a[2];
656  So2<Scalar> so2 = So2<Scalar>::exp(theta);
657  Scalar sin_theta_by_theta;
658  Scalar one_minus_cos_theta_by_theta;
659  using std::abs;
660 
661  if (abs(theta) < kEpsilon<Scalar>) {
662  Scalar theta_sq = theta * theta;
663  sin_theta_by_theta = Scalar(1.) - Scalar(1. / 6.) * theta_sq;
664  one_minus_cos_theta_by_theta =
665  Scalar(0.5) * theta - Scalar(1. / 24.) * theta * theta_sq;
666  } else {
667  sin_theta_by_theta = so2.unitComplex().y() / theta;
668  one_minus_cos_theta_by_theta =
669  (Scalar(1.) - so2.unitComplex().x()) / theta;
670  }
671  Eigen::Vector2<Scalar> trans(
672  sin_theta_by_theta * vec_a[0] - one_minus_cos_theta_by_theta * vec_a[1],
673  one_minus_cos_theta_by_theta * vec_a[0] +
674  sin_theta_by_theta * vec_a[1]);
675  return Se2<Scalar>(so2, trans);
676  }

679  ///
680  template <class TS = Scalar>
681  static SOPHUS_FUNC std::enable_if_t<std::is_floating_point<TS>::value, Se2>
682  fitToSe2(Eigen::Matrix3<Scalar> const& mat3x3) {

685  mat3x3.template block<2, 1>(0, 2));

688  /// Returns the ith infinitesimal generators of SE(2).
689  ///
690  /// The infinitesimal generators of SE(2) are:
691  ///
692  /// ```
693  /// | 0 0 1 |
694  /// G_0 = | 0 0 0 |
695  /// | 0 0 0 |
696  ///
697  /// | 0 0 0 |
698  /// G_1 = | 0 0 1 |
699  /// | 0 0 0 |
700  ///
701  /// | 0 -1 0 |
702  /// G_2 = | 1 0 0 |
703  /// | 0 0 0 |
704  /// ```
705  ///
706  /// Precondition: ``i`` must be in 0, 1 or 2.
707  ///
708  SOPHUS_FUNC static Transformation generator(int i) {
709  SOPHUS_ASSERT(i >= 0 || i <= 2, "i should be in range [0,2].");
710  Tangent e;
711  e.setZero();
712  e[i] = Scalar(1);
713  return hat(e);
714  }

716  /// hat-operator
717  ///
718  /// It takes in the 3-vector representation (= twist) and returns the
719  /// corresponding matrix representation of Lie algebra element.
720  ///
721  /// Formally, the hat()-operator of SE(3) is defined as
722  ///
723  /// ``hat(.): R^3 -> R^{3x33}, hat(a) = sum_i a_i * G_i`` (for i=0,1,2)
724  ///
725  /// with ``G_i`` being the ith infinitesimal generator of SE(2).
726  ///
727  /// The corresponding inverse is the vee()-operator, see below.
728  ///
729  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
730  Transformation mat_omega;
731  mat_omega.setZero();
732  mat_omega.template topLeftCorner<2, 2>() = So2<Scalar>::hat(vec_a[2]);
733  mat_omega.col(2).template head<2>() = vec_a.template head<2>();
734  return mat_omega;
735  }
736 
737  /// Lie bracket
738  ///
739  /// It computes the Lie bracket of SE(2). To be more specific, it computes
740  ///
741  /// ``[omega_1, omega_2]_se2 := vee([hat(omega_1), hat(omega_2)])``
742  ///
743  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
744  /// hat()-operator and ``vee(.)`` the vee()-operator of SE(2).
745  ///
746  SOPHUS_FUNC static Tangent lieBracket(
747  Tangent const& vec_a, Tangent const& vec_b) {
748  Eigen::Vector2<Scalar> upsilon1 = vec_a.template head<2>();
749  Eigen::Vector2<Scalar> upsilon2 = vec_b.template head<2>();
750  Scalar theta1 = vec_a[2];
751  Scalar theta2 = vec_b[2];
752 
753  return Tangent(
754  -theta1 * upsilon2[1] + theta2 * upsilon1[1],
755  theta1 * upsilon2[0] - theta2 * upsilon1[0],
756  Scalar(0));
757  }

800  /// vee-operator
801  ///
802  /// It takes the 3x3-matrix representation ``Omega`` and maps it to the
803  /// corresponding 3-vector representation of Lie algebra.
804  ///
805  /// This is the inverse of the hat()-operator, see above.
806  ///
807  /// Precondition: ``Omega`` must have the following structure:
808  ///
809  /// | 0 -d a |
810  /// | d 0 b |
811  /// | 0 0 0 |
812  ///
813  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
814  SOPHUS_ASSERT(
815  omega.row(2).template lpNorm<1>() < kEpsilon<Scalar>,
816  "Omega: \n{}",
817  omega);
818  Tangent upsilon_omega;
819  upsilon_omega.template head<2>() = omega.col(2).template head<2>();
820  upsilon_omega[2] = So2<Scalar>::vee(omega.template topLeftCorner<2, 2>());
821  return upsilon_omega;
822  }
