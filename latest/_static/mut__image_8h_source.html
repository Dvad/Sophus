35 using UniqueDataArea = std::unique_ptr<uint8_t, UniqueDataAreaDeleter>;

37 /// A image with write access to pixels and exclusive ownership. There is no
38 /// copy constr / copy assignment, but move constr / assignment.
39 ///
40 /// Content from a MutImage can be moved into an Image.
41 ///
42 /// Type is nullable. In that case `this->isEmpty()` is true.
43 ///
44 /// Similar to Pangolin::ManagedImage.
45 template <
46  class TPixel,
47  template <class> class TAllocator = Eigen::aligned_allocator>
48 class MutImage : public MutImageView<TPixel> {
49  public:
50  struct TypedDeleterImpl {
51  TypedDeleterImpl(size_t num_bytes) : num_bytes(num_bytes) {}
52 
53  void operator()(TPixel* p) const {
54  if (p != nullptr) {
55  TAllocator<TPixel>().deallocate(p, num_bytes / sizeof(TPixel));
56  }
57  }
58 
59  size_t num_bytes;
60  };
61 
62  /// Deleter for MutImage.
63  struct Deleter : UniqueDataAreaDeleter {
64  Deleter() = default;
65  Deleter(TypedDeleterImpl image_deleter) : image_deleter(image_deleter) {}
66 
67  virtual void operator()(uint8_t* p) const final {
68  if (image_deleter) {
69  (*image_deleter)(reinterpret_cast<TPixel*>(p));
70  }
71  }
72 
73  std::optional<TypedDeleterImpl> image_deleter;
74  };

77  friend class Image;

79  /// Constructs empty image.

82  /// Creates new image with given shape.
83  ///
84  /// If shape is not empty, memory allocation will happen.
85  explicit MutImage(ImageShape shape)
86  : MutImageView<TPixel>(shape, nullptr), unique_(nullptr) {
87  if (shape.sizeBytes() != 0u) {
88  SOPHUS_ASSERT_EQ(shape.sizeBytes() % sizeof(TPixel), 0);
89  this->unique_ = UniqueDataArea(
90  (uint8_t*)(TAllocator<TPixel>().allocate(
91  shape.sizeBytes() / sizeof(TPixel))),
92  Deleter(TypedDeleterImpl(shape.sizeBytes())));
93  }
94  this->ptr_ = reinterpret_cast<TPixel*>(this->unique_.get());
95  }
96

98  ///
99  /// If shape is not empty, memory allocation will happen.
100  explicit MutImage(sophus::ImageSize size)
101  : MutImage(ImageShape::makeFromSize<TPixel>(size)) {}

103  /// Creates contiguous copy from view.
104  ///
105  /// If view is not empty, memory allocation will happen.
106  [[nodiscard]] static MutImage makeCopyFrom(ImageView<TPixel> const& view) {
107  MutImage image(view.imageSize());
108  image.copyDataFrom(view);

110  }

112  /// Creates new MutImage given view and unary transform function.
113  ///
114  /// mut_image(u, v) = unary_op(view(u, v));
115  template <class TOtherPixel, class TUnaryOperation>
116  static MutImage makeFromTransform(
117  ImageView<TOtherPixel> view, TUnaryOperation const& unary_op) {
118  MutImage mut_image(view.imageSize());
119  mut_image.transformFrom(view, unary_op);
120  return mut_image;
121  }
122 
123  /// Creates new MutImage given two views and binary transform function.
124  ///
125  /// mut_image(u, v) = binary_op(lhs(u, v), rhs(u, v));
126  template <class TLhsPixel, class TRhsPixel, class TBinaryOperation>
127  static MutImage makeFromTransform(
128  ImageView<TLhsPixel> lhs,
129  ImageView<TRhsPixel> rhs,
130  TBinaryOperation const& binary_op) {
131  MutImage mut_image(lhs.imageSize());
132  mut_image.transformFrom(lhs, rhs, binary_op);
133  return mut_image;
134  }

148  /// Move constructor - is cheap - no memory allocations.
149  MutImage(MutImage&& img) noexcept
150  : MutImageView<TPixel>(img.viewMut()), unique_(std::move(img.unique_)) {
151  this->ptr_ = reinterpret_cast<TPixel*>(unique_.get());
152  img.setViewToEmpty();
153  }
154 
155  /// Move assignment
156  MutImage& operator=(MutImage&& img) noexcept {
157  reset();
158  this->shape_ = img.shape_;
159  this->unique_ = std::move(img.unique_);
160  this->ptr_ = reinterpret_cast<TPixel*>(unique_.get());
161  img.setViewToEmpty();
162  return *this;
163  }
164  // End (Rule of 5)
165 
166  [[nodiscard]] MutImageView<TPixel> viewMut() const {
167  return MutImageView<TPixel>(this->shape(), this->ptrMut());
168  }
169 
170  /// Swaps img and this.
171  void swap(MutImage& img) {
172  std::swap(img.shape_, this->shape_);
173  std::swap(img.ptr_, this->ptr_);
174  std::swap(img.unique_, this->unique_);
175  }
176 
177  /// Clears image.
178  ///
179  /// If image was not empty, memory deallocations will happen.
180  void reset() {
181  unique_.reset();
182  this->setViewToEmpty();
183  }
184 
185  protected:
186  /// Leaks memory and returns deleter.
187  Deleter leakAndReturnDeleter() {
188  SOPHUS_ASSERT(!this->isEmpty());
189  Deleter& del = *std::get_deleter<Deleter>(unique_);
190  Deleter del_copy = del;
191  del.image_deleter.reset();
192  return del_copy;
193  }
194 
195  /// MutImage has unique ownership.
196  UniqueDataArea unique_; // NOLINT
197 };
