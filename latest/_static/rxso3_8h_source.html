58 /// RxSo3 base type - implements RxSo3 class but is storage agnostic
59 ///
60 /// This class implements the group ``R+ x SO(3)``, the direct product of the
61 /// group of positive scalar 3x3 matrices (= isomorph to the positive
62 /// real numbers) and the three-dimensional special orthogonal group SO(3).
63 /// Geometrically, it is the group of rotation and scaling in three dimensions.
64 /// As a matrix groups, RxSo3 consists of matrices of the form ``s * R``
65 /// where ``R`` is an orthogonal matrix with ``det(R)=1`` and ``s > 0``
66 /// being a positive real number.
67 ///
68 /// Internally, RxSo3 is represented by the group of non-zero quaternions.
69 /// In particular, the scale equals the squared(!) norm of the quaternion ``q``,
70 /// ``s = |q|^2``. This is a most compact representation since the degrees of
71 /// freedom (kDoF) of RxSo3 (=4) equals the number of internal parameters (=4).
72 ///
73 /// This class has the explicit class invariant that the scale ``s`` is not
74 /// too close to either zero or infinity. Strictly speaking, it must hold that:
75 ///
76 /// ``quaternion().squaredNorm() >= Constants::epsilon()`` and
77 /// ``1. / quaternion().squaredNorm() >= Constants::epsilon()``.
78 ///
79 /// In order to obey this condition, group multiplication is implemented with
80 /// saturation such that a product always has a scale which is equal or greater
81 /// this threshold.
82 template <class TDerived>
83 class RxSo3Base {
84  public:
85  static int constexpr kOptions = Eigen::internal::traits<TDerived>::kOptions;
86  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
87  using QuaternionType =
88  typename Eigen::internal::traits<TDerived>::QuaternionType;
89  using QuaternionTemporaryType = Eigen::Quaternion<Scalar, kOptions>;

132  /// Adjoint transformation
133  ///
134  /// This function return the adjoint transformation ``Ad`` of the group
135  /// element ``A`` such that for all ``x`` it holds that
136  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
137  ///
138  /// For RxSO(3), it simply returns the rotation matrix corresponding to ``A``.
139  ///
140  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
141  Adjoint res;
142  res.setIdentity();
143  res.template topLeftCorner<3, 3>() = rotationMatrix();
144  return res;
145  }

176  /// Logarithmic map
177  ///
178  /// Computes the logarithm, the inverse of the group exponential which maps
179  /// element of the group (scaled rotation matrices) to elements of the tangent
180  /// space (rotation-vector plus logarithm of scale factor).
181  ///
182  /// To be specific, this function computes ``vee(logmat(.))`` with
183  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
184  /// of RxSo3.
185  ///
186  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
187  return logAndTheta().tangent;
188  }
189 
190  /// As above, but also returns ``theta = |omega|``.
191  ///
192  SOPHUS_FUNC [[nodiscard]] TangentAndTheta logAndTheta() const {
193  using std::log;
194 
195  Scalar scale = quaternion().squaredNorm();
196  TangentAndTheta result;
197  result.tangent[3] = log(scale);
198  auto omega_and_theta = So3<Scalar>(quaternion()).logAndTheta();
199  result.tangent.template head<3>() = omega_and_theta.tangent;
200  result.theta = omega_and_theta.theta;
201  return result;
202  }
203  /// Returns 3x3 matrix representation of the instance.
204  ///
205  /// For RxSo3, the matrix representation is an scaled orthogonal matrix ``sR``
206  /// with ``det(R)=s^3``, thus a scaled rotation matrix ``R`` with scale
207  /// ``s``.
208  ///
209  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
210  Transformation s_r;
211 
212  Scalar const vx_sq = quaternion().vec().x() * quaternion().vec().x();
213  Scalar const vy_sq = quaternion().vec().y() * quaternion().vec().y();
214  Scalar const vz_sq = quaternion().vec().z() * quaternion().vec().z();
215  Scalar const w_sq = quaternion().w() * quaternion().w();
216  Scalar const two_vx = Scalar(2) * quaternion().vec().x();
217  Scalar const two_vy = Scalar(2) * quaternion().vec().y();
218  Scalar const two_vz = Scalar(2) * quaternion().vec().z();
219  Scalar const two_vx_vy = two_vx * quaternion().vec().y();
220  Scalar const two_vx_vz = two_vx * quaternion().vec().z();
221  Scalar const two_vx_w = two_vx * quaternion().w();
222  Scalar const two_vy_vz = two_vy * quaternion().vec().z();
223  Scalar const two_vy_w = two_vy * quaternion().w();
224  Scalar const two_vz_w = two_vz * quaternion().w();
225 
226  s_r(0, 0) = vx_sq - vy_sq - vz_sq + w_sq;
227  s_r(1, 0) = two_vx_vy + two_vz_w;
228  s_r(2, 0) = two_vx_vz - two_vy_w;
229 
230  s_r(0, 1) = two_vx_vy - two_vz_w;
231  s_r(1, 1) = -vx_sq + vy_sq - vz_sq + w_sq;
232  s_r(2, 1) = two_vx_w + two_vy_vz;
233 
234  s_r(0, 2) = two_vx_vz + two_vy_w;
235  s_r(1, 2) = -two_vx_w + two_vy_vz;
236  s_r(2, 2) = -vx_sq - vy_sq + vz_sq + w_sq;
237  return s_r;
238  }

249  /// Group multiplication, which is rotation concatenation and scale
250  /// multiplication.
251  ///
252  /// Note: This function performs saturation for products close to zero in
253  /// order to ensure the class invariant.
254  ///
255  template <class TOtherDerived>
256  SOPHUS_FUNC RxSo3Product<TOtherDerived> operator*(
257  RxSo3Base<TOtherDerived> const& other) const {
258  using std::sqrt;
259  using ResultT = ReturnScalar<TOtherDerived>;
260  using QuaternionProductType =
261  typename RxSo3Product<TOtherDerived>::QuaternionType;
262 
263  QuaternionProductType result_quaternion(
264  sophus::So3<double>::quaternionProduct<QuaternionProductType>(
265  quaternion(), other.quaternion()));
266 
267  ResultT scale = result_quaternion.squaredNorm();
268  if (scale < kEpsilon<ResultT>) {
269  SOPHUS_ASSERT(scale > ResultT(0), "Scale must be greater zero.");
270  /// Saturation to ensure class invariant.
271  result_quaternion.normalize();
272  result_quaternion.coeffs() *= sqrt(kEpsilonPlus<ResultT>);
273  }
274  if (scale > ResultT(1.) / kEpsilon<ResultT>) {
275  result_quaternion.normalize();
276  result_quaternion.coeffs() /= sqrt(kEpsilonPlus<ResultT>);
277  }
278  return RxSo3Product<TOtherDerived>(result_quaternion);
279  }

281  /// Group action on 3-points.
282  ///
283  /// This function rotates a 3 dimensional point ``p`` by the So3 element
284  /// ``bar_R_foo`` (= rotation matrix) and scales it by the scale factor
285  /// ``s``:
286  ///
287  /// ``p_bar = s * (bar_R_foo * p_foo)``.
288  ///
289  template <
290  typename TPointDerived,
291  typename = typename std::enable_if<
292  IsFixedSizeVector<TPointDerived, 3>::value>::type>
293  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
294  Eigen::MatrixBase<TPointDerived> const& p) const {
295  // Follows http:///eigen.tuxfamily.org/bz/show_bug.cgi?id=459
296  Scalar scale = quaternion().squaredNorm();
297  PointProduct<TPointDerived> two_vec_cross_p = quaternion().vec().cross(p);
298  two_vec_cross_p += two_vec_cross_p;
299  return scale * p + (quaternion().w() * two_vec_cross_p +
300  quaternion().vec().cross(two_vec_cross_p));
301  }
302 
303  /// Group action on homogeneous 3-points. See above for more details.
304  ///
305  template <
306  typename THPointDerived,
307  typename = typename std::enable_if<
308  IsFixedSizeVector<THPointDerived, 4>::value>::type>
309  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
310  Eigen::MatrixBase<THPointDerived> const& p) const {
311  auto const rsp = *this * p.template head<3>();
312  return HomogeneousPointProduct<THPointDerived>(
313  rsp(0), rsp(1), rsp(2), p(3));
314  }

316  /// Group action on lines.
317  ///
318  /// This function rotates a parametrized line ``l(t) = o + t * d`` by the So3
319  /// element ans scales it by the scale factor:
320  ///
321  /// Origin ``o`` is rotated and scaled
322  /// Direction ``d`` is rotated (preserving it's norm)
323  ///
324  SOPHUS_FUNC Line operator*(Line const& l) const {
325  return Line(
326  (*this) * l.origin(),
327  (*this) * l.direction() / quaternion().squaredNorm());
328  }
329 
330  /// Group action on planes.
331  ///
332  /// This function rotates parametrized plane
333  /// ``n.x + d = 0`` by the So3 element and scales it by the scale factor:
334  ///
335  /// Normal vector ``n`` is rotated
336  /// Offset ``d`` is scaled
337  ///
338  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
339  auto const this_scale = scale();
340  return Hyperplane(
341  (*this) * p.normal() / this_scale, this_scale * p.offset());
342  }

424  /// Setter of quaternion using scaled rotation matrix ``sR``.
425  ///
426  /// Precondition: The 3x3 matrix must be "scaled orthogonal"
427  /// and have a positive determinant.
428  ///
429  SOPHUS_FUNC void setScaledRotationMatrix(Transformation const& s_r) {
430  Transformation squared_s_r = s_r * s_r.transpose();
431  Scalar squared_scale =
432  Scalar(1. / 3.) *
433  (squared_s_r(0, 0) + squared_s_r(1, 1) + squared_s_r(2, 2));
434  SOPHUS_ASSERT(
435  squared_scale >= kEpsilon<Scalar> * kEpsilon<Scalar>,
436  "Scale factor must be greater-equal epsilon.");
437  SOPHUS_ASSERT(
438  squared_scale < Scalar(1.) / (kEpsilon<Scalar> * kEpsilon<Scalar>),
439  "Inverse scale factor must be greater-equal epsilon.");
440  Scalar scale = sqrt(squared_scale);
441  quaternionNonconst() = s_r / scale;
442  quaternionNonconst().coeffs() *= sqrt(scale);
443  }

458  /// Returns derivative of this * RxSo3::exp(x) wrt. x at x=0
459  ///
460  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>
461  dxThisMulExpXAt0() const {
462  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
463  Eigen::Quaternion<Scalar> const q = quaternion();
464  j.col(3) = q.coeffs() * Scalar(0.5);
465  Scalar const c0 = Scalar(0.5) * q.w();
466  Scalar const c1 = Scalar(0.5) * q.z();
467  Scalar const c2 = -c1;
468  Scalar const c3 = Scalar(0.5) * q.y();
469  Scalar const c4 = Scalar(0.5) * q.x();
470  Scalar const c5 = -c4;
471  Scalar const c6 = -c3;
472  j(0, 0) = c0;
473  j(0, 1) = c2;
474  j(0, 2) = c3;
475  j(1, 0) = c1;
476  j(1, 1) = c0;
477  j(1, 2) = c5;
478  j(2, 0) = c6;
479  j(2, 1) = c4;
480  j(2, 2) = c0;
481  j(3, 0) = c5;
482  j(3, 1) = c6;
483  j(3, 2) = c2;
484  return j;
485  }
486 
487  /// Returns derivative of log(this^{-1} * x) by x at x=this.
488  ///
489  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>
490  dxLogThisInvTimesXAtThis() const {
491  auto& q = quaternion();
492  Eigen::Matrix<Scalar, kDoF, kNumParameters> j;
493  // clang-format off
494  j << q.w(), q.z(), -q.y(), -q.x(),
495  -q.z(), q.w(), q.x(), -q.y(),
496  q.y(), -q.x(), q.w(), -q.z(),
497  q.x(), q.y(), q.z(), q.w();
498  // clang-format on
499  const Scalar scaler = Scalar(2.) / q.squaredNorm();
500  return j * scaler;
501  }

643  /// Returns derivative of exp(x) wrt. x.
644  ///
645  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
646  Tangent const& a) {
647  using std::exp;
648  using std::sqrt;
649  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
650  Eigen::Vector3<Scalar> const omega = a.template head<3>();
651  Scalar const sigma = a[3];
652  Eigen::Quaternion<Scalar> quat = So3<Scalar>::exp(omega).unitQuaternion();
653  Scalar const scale = sqrt(exp(sigma));
654  Scalar const scale_half = scale * Scalar(0.5);
655 
656  j.template block<4, 3>(0, 0) = So3<Scalar>::dxExpX(omega) * scale;
657  j.col(3) = quat.coeffs() * scale_half;
658  return j;
659  }

662  ///

664  Point const& point) {
665  Eigen::Matrix<Scalar, 3, kDoF> j;

670  /// Returns derivative of ``expmat(x)`` wrt. ``x_i at x=0``, with
671  /// ``expmat(.)`` being the matrix exponential.
672  ///
673  SOPHUS_FUNC static Transformation dxiExpmatXAt0(int i) {
674  return generator(i);
675  }
676  /// Group exponential
677  ///
678  /// This functions takes in an element of tangent space (= rotation 3-vector
679  /// plus logarithm of scale) and returns the corresponding element of the
680  /// group RxSo3.
681  ///
682  /// To be more specific, thixs function computes ``expmat(hat(omega))``
683  /// with ``expmat(.)`` being the matrix exponential and ``hat(.)`` being the
684  /// hat()-operator of RSO3.
685  ///
686  SOPHUS_FUNC static RxSo3<Scalar> exp(Tangent const& a) {
687  return expAndTheta(a).rxso3;
688  }
689 
690  /// As above, but also returns ``theta = |omega|``.
691  ///
692  SOPHUS_FUNC static RxSo3AndTheta expAndTheta(Tangent const& vec_a) {
693  using std::exp;
694  using std::max;
695  using std::min;
696  using std::sqrt;
697 
698  RxSo3AndTheta rxso3_and_theta;
699 
700  Eigen::Vector3<Scalar> const vec_omega = vec_a.template head<3>();
701  Scalar sigma = vec_a[3];
702  Scalar scale = exp(sigma);
703  // Ensure that scale-factor contraint is always valid
704  scale = max(scale, kEpsilonPlus<Scalar>);
705  scale = min(scale, Scalar(1.) / kEpsilonPlus<Scalar>);
706  Scalar sqrt_scale = sqrt(scale);
707  auto so3_and_theta = So3<Scalar>::expAndTheta(vec_omega);
708  Eigen::Quaternion<Scalar> quat = so3_and_theta.so3.unitQuaternion();
709  quat.coeffs() *= sqrt_scale;
710  rxso3_and_theta.rxso3 = RxSo3<Scalar>(quat);
711  rxso3_and_theta.theta = so3_and_theta.theta;
712  return rxso3_and_theta;
713  }
714 
715  /// Returns the ith infinitesimal generators of ``R+ x SO(3)``.
716  ///
717  /// The infinitesimal generators of RxSo3 are:
718  ///
719  /// ```
720  /// | 0 0 0 |
721  /// G_0 = | 0 0 -1 |
722  /// | 0 1 0 |
723  ///
724  /// | 0 0 1 |
725  /// G_1 = | 0 0 0 |
726  /// | -1 0 0 |
727  ///
728  /// | 0 -1 0 |
729  /// G_2 = | 1 0 0 |
730  /// | 0 0 0 |
731  ///
732  /// | 1 0 0 |
733  /// G_2 = | 0 1 0 |
734  /// | 0 0 1 |
735  /// ```
736  ///
737  /// Precondition: ``i`` must be 0, 1, 2 or 3.
738  ///
739  SOPHUS_FUNC static Transformation generator(int i) {
740  SOPHUS_ASSERT(i >= 0 && i <= 3, "i should be in range [0,3].");
741  Tangent e;
742  e.setZero();
743  e[i] = Scalar(1);
744  return hat(e);
745  }
746 
747  /// hat-operator
748  ///
749  /// It takes in the 4-vector representation ``a`` (= rotation vector plus
750  /// logarithm of scale) and returns the corresponding matrix representation
751  /// of Lie algebra element.
752  ///
753  /// Formally, the hat()-operator of RxSo3 is defined as
754  ///
755  /// ``hat(.): R^4 -> R^{3x3}, hat(a) = sum_i a_i * G_i`` (for i=0,1,2,3)
756  ///
757  /// with ``G_i`` being the ith infinitesimal generator of RxSo3.
758  ///
759  /// The corresponding inverse is the vee()-operator, see below.
760  ///
761  SOPHUS_FUNC static Transformation hat(Tangent const& a) {
762  Transformation mat_a;
763  // clang-format off
764  mat_a << a(3), -a(2), a(1),
765  a(2), a(3), -a(0),
766  -a(1), a(0), a(3);
767  // clang-format on
768  return mat_a;
769  }
770 
771  /// Lie bracket
772  ///
773  /// It computes the Lie bracket of RxSO(3). To be more specific, it computes
774  ///
775  /// ``[omega_1, omega_2]_rxso3 := vee([hat(omega_1), hat(omega_2)])``
776  ///
777  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
778  /// hat()-operator and ``vee(.)`` the vee()-operator of RxSo3.
779  ///
780  SOPHUS_FUNC static Tangent lieBracket(Tangent const& a, Tangent const& b) {
781  Eigen::Vector3<Scalar> const omega1 = a.template head<3>();
782  Eigen::Vector3<Scalar> const omega2 = b.template head<3>();
783  Eigen::Vector4<Scalar> res;
784  res.template head<3>() = omega1.cross(omega2);
785  res[3] = Scalar(0);
786  return res;
787  }
788 
789  /// Draw uniform sample from RxSO(3) manifold.
790  ///
791  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
792  /// hence the scale is in [0.5, 2].
793  ///
794  template <class TUniformRandomBitGenerator>
795  static RxSo3 sampleUniform(TUniformRandomBitGenerator& generator) {
796  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
797  using std::exp2;
798  return RxSo3(
799  exp2(uniform(generator)), So3<Scalar>::sampleUniform(generator));
800  }
801 
802  /// vee-operator
803  ///
804  /// It takes the 3x3-matrix representation ``Omega`` and maps it to the
805  /// corresponding vector representation of Lie algebra.
806  ///
807  /// This is the inverse of the hat()-operator, see above.
808  ///
809  /// Precondition: ``Omega`` must have the following structure:
810  ///
811  /// | d -c b |
812  /// | c d -a |
813  /// | -b a d |
814  ///
815  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
816  using std::abs;
817  return Tangent(omega(2, 1), omega(0, 2), omega(1, 0), omega(0, 0));
818  }

822 
823  QuaternionMember quaternion_; // NOLINT
824 };
