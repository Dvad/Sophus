19 template <class T>
20 concept LieGroupImplConcept =
21  (T::kPointDim == 2 || T::kPointDim == 3) && // 2d or 3d points
22  (T::kPointDim == T::kAmbientDim // inhomogeneous point representation
23  ||
24  T::kPointDim + 1 == T::kAmbientDim) // or homogeneous point representation
25  && requires(
26  T g,
27  Eigen::Vector<typename T::Scalar, T::kDof> tangent,
28  Eigen::Vector<typename T::Scalar, T::kPointDim> point,
29  Eigen::Vector<typename T::Scalar, T::kNumParams> params) {
30  {
31  T::exp(tangent)
32  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kNumParams>>;
33 
34  {
35  T::log(params)
36  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kDof>>;
37 
38  {
39  T::inverse(params)
40  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kNumParams>>;
41 
42  {
43  T::action(params, point)
44  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kPointDim>>;
45 
46  {
47  T::multiplication(params, params)
48  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kNumParams>>;
49 
50  {
51  T::matrix(params)
52  } -> ConvertibleTo<
53  Eigen::Matrix<typename T::Scalar, T::kPointDim, T::kAmbientDim>>;
54 
55  {
56  T::homogeneousMatrix(params)
57  } -> ConvertibleTo<
58  Eigen::Matrix<typename T::Scalar, T::kAmbientDim, T::kAmbientDim>>;
59 
60  {
61  T::exampleTangents()
62  } -> ConvertibleTo<std::vector<Eigen::Vector<typename T::Scalar, T::kDof>>>;
63 
64  {
65  T::exampleParams()
66  } -> ConvertibleTo<
67  std::vector<Eigen::Vector<typename T::Scalar, T::kNumParams>>>;
68 };

84 template <
85  class TScalar,
86  template <class>
87  class TLeftGroup,
88  template <class>
89  class TRightGroup>
90 requires LieGroupImplConcept<TLeftGroup<TScalar>> &&
91  LieGroupImplConcept<TRightGroup<TScalar>>
92 class DirectProduct {
93  public:
94  using Scalar = TScalar;
95  using LeftGroup = TLeftGroup<Scalar>;
96  using RightGroup = TRightGroup<Scalar>;
97 
98  static int const kDof = LeftGroup::kDof + RightGroup::kDof;
99  static int const kNumParams = LeftGroup::kNumParams + RightGroup::kNumParams;
100  static int const kPointDim = LeftGroup::kPointDim;
101  static_assert(kPointDim == RightGroup::kPointDim);
102  static int const kAmbientDim = LeftGroup::kAmbientDim;
103  static_assert(kAmbientDim == RightGroup::kAmbientDim);
104  static_assert(kPointDim == kAmbientDim);
105 
106  static auto exp(Eigen::Vector<Scalar, kDof> tangent)
107  -> Eigen::Vector<Scalar, kNumParams> {
108  return params(
109  LeftGroup::exp(leftTangent(tangent)),
110  RightGroup::exp(rightTangent(tangent)));
111  }
112 
113  static auto log(Eigen::Vector<Scalar, kNumParams> const& params)
114  -> Eigen::Vector<Scalar, kDof> {
115  return tangent(
116  LeftGroup::log(leftParams(params)),
117  RightGroup::log(rightParams(params)));
118  }
119 
120  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& params)
121  -> Eigen::Vector<Scalar, kNumParams> {
122  return params(
123  LeftGroup::inverse(leftParams(params)),
124  RightGroup::inverse(rightParams(params)));
125  }
126 
127  static auto multiplication(
128  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
129  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
130  -> Eigen::Vector<Scalar, kNumParams> {
131  return params(
132  LeftGroup::multiplication(
133  leftParams(lhs_params), leftParams(rhs_params)),
134  RightGroup::multiplication(
135  rightParams(lhs_params), rightParams(rhs_params)));
136  }
137 
138  static auto action(
139  Eigen::Vector<Scalar, kNumParams> const& params,
140  Eigen::Vector<Scalar, kPointDim> const& point)
141  -> Eigen::Vector<Scalar, kPointDim> {
142  return LeftGroup::action(RightGroup::action(point));
143  }
144 
145  static auto matrix(Eigen::Vector<Scalar, kNumParams> const& params)
146  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
147  return LeftGroup::matrix(leftParams(params)) *
148  RightGroup::matrix(rightParams(params));
149  }
150 
151  static auto homogeneousMatrix(Eigen::Vector<Scalar, kNumParams> const& params)
152  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
153  return matrix(params);
154  }
155 
156  static auto leftJacobian(Eigen::Vector<Scalar, kNumParams> const&)
157  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
158  return LeftGroup::leftJacobian(leftParams(params)) *
159  RightGroup::leftJacobian(rightParams(params));
160  }
161 
162  static auto leftJacobianInverse(Eigen::Vector<Scalar, kNumParams> const&)
163  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
164  return LeftGroup::leftJacobianInverse(leftParams(params)) *
165  RightGroup::leftJacobianInverse(rightParams(params));
166  }
167 
168  static auto exampleTangents() -> std::vector<Eigen::Vector<Scalar, kDof>> {
169  std::vector<Eigen::Vector<Scalar, kDof>> examples;
170  for (auto const& left_tangent : LeftGroup::exampleTangents()) {
171  for (auto const& right_tangents : LeftGroup::exampleTangents()) {
172  examples.push_back(tagent(left_tangent, right_tangents));
173  }
174  }
175  return examples;
176  }
177 
178  static auto exampleParams()
179  -> std::vector<Eigen::Vector<Scalar, kNumParams>> {
180  std::vector<Eigen::Vector<Scalar, kNumParams>> examples;
181  for (auto const& left_params : LeftGroup::exampleParams()) {
182  for (auto const& right_params : LeftGroup::exampleParams()) {
183  examples.push_back(params(left_params, right_params));
184  }
185  }
186  return examples;
187  }
188 
189  private:
190  static auto leftParams(Eigen::Vector<Scalar, kNumParams> const& params)
191  -> Eigen::Vector<Scalar, LeftGroup::kNumParams> {
192  return params.template head<LeftGroup::kNumParams>();
193  }
194 
195  static auto rightParams(Eigen::Vector<Scalar, kNumParams> const& params)
196  -> Eigen::Vector<Scalar, RightGroup::kNumParams> {
197  return params.template tail<RightGroup::kNumParams>();
198  }
199 
200  static auto params(
201  Eigen::Vector<Scalar, LeftGroup::kNumParams> const& params1,
202  Eigen::Vector<Scalar, RightGroup::kNumParams> const& params2)
203  -> Eigen::Vector<Scalar, kNumParams> {
204  Eigen::Vector<Scalar, kNumParams> params;
205  params.template head<LeftGroup::kNumParams>() = params1;
206  params.template tail<RightGroup::kNumParams>() = params2;
207  return params;
208  }
209 
210  static auto leftTangent(Eigen::Vector<Scalar, kDof> const& tangent)
211  -> Eigen::Vector<Scalar, LeftGroup::kDof> {
212  return tangent.template head<LeftGroup::kDof>();
213  }
214 
215  static auto rightTangent(Eigen::Vector<Scalar, kDof> const& tangent)
216  -> Eigen::Vector<Scalar, RightGroup::kDof> {
217  return tangent.template tail<RightGroup::kDof>();
218  }
219 
220  static auto tangent(
221  Eigen::Vector<Scalar, LeftGroup::kDof> const& tangent1,
222  Eigen::Vector<Scalar, RightGroup::kDof> const& tangent2)
223  -> Eigen::Vector<Scalar, kDof> {
224  Eigen::Vector<Scalar, kDof> tangent;
225  tangent.template head<LeftGroup::kDof>() = tangent1;
226  tangent.template tail<RightGroup::kDof>() = tangent2;
227  return tangent;
228  }

231 template <class TScalar>
232 class Rotation2 {
233  public:
234  using Scalar = TScalar;
235  static int const kDof = 1;
236  static int const kNumParams = 2;
237  static int const kPointDim = 2;
238  static int const kAmbientDim = 2;
239 
240  static auto exp(Eigen::Vector<Scalar, kDof> const& angle)
241  -> Eigen::Vector<Scalar, kNumParams> {
242  using std::cos;
243  using std::sin;
244  Eigen::Vector<Scalar, 2>(cos(angle[0]), sin(angle[0]));
245  }
246 
247  static auto log(Eigen::Vector<Scalar, kNumParams> const& unit_complex)
248  -> Eigen::Vector<Scalar, kDof> {
249  using std::atan2;
250  return Eigen::Vector<Scalar, kDof>({unit_complex.y(), unit_complex.x()});
251  }
252 
253  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& unit_complex)
254  -> Eigen::Vector<Scalar, kNumParams> {
255  return Eigen::Vector<Scalar, kNumParams>(
256  unit_complex.x(), -unit_complex.y());
257  }
258 
259  static auto action(
260  Eigen::Vector<Scalar, kNumParams> const& params,
261  Eigen::Vector<Scalar, kPointDim> const& point)
262  -> Eigen::Vector<Scalar, kPointDim>;
263 
264  static auto multiplication(
265  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
266  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
267  -> Eigen::Vector<Scalar, kNumParams> {
268  Scalar const lhs_real = lhs_params.x();
269  Scalar const lhs_imag = lhs_params.y();
270  Scalar const rhs_real = rhs_params.x();
271  Scalar const rhs_imag = rhs_params.y();
272 
273  // complex multiplication
274  Scalar const result_real = lhs_real * rhs_real - lhs_imag * rhs_imag;
275  Scalar const result_imag = lhs_real * rhs_imag + lhs_imag * rhs_real;
276 
277  Scalar const squared_norm =
278  result_real * result_real + result_imag * result_imag;
279  // We can assume that the squared-norm is close to 1 since we deal with a
280  // unit complex number. Due to numerical precision issues, there might
281  // be a small drift after pose concatenation. Hence, we need to renormalizes
282  // the complex number here.
283  // Since squared-norm is close to 1, we do not need to calculate the costly
284  // square-root, but can use an approximation around 1 (see
285  // http://stackoverflow.com/a/12934750 for details).
286  if (squared_norm != 1.0) {
287  Scalar const scale = 2.0 / (1.0 + squared_norm);
288  return Eigen::Vector<Scalar, kNumParams>(
289  result_real * scale, result_imag * scale);
290  }
291  return Eigen::Vector<Scalar, kNumParams>(result_real, result_imag);
292  }
293 
294  static auto matrix(Eigen::Vector<Scalar, kNumParams> const& params)
295  -> Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim>;
296 
297  static auto homogeneousMatrix(Eigen::Vector<Scalar, kNumParams> const& params)
298  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
299  return matrix(params);
300  }
301 
302  static auto leftJacobian(Eigen::Vector<Scalar, kNumParams> const&)
303  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
304  return Eigen::Matrix<Scalar, kPointDim, kPointDim>::Identity();
305  }
306 
307  static auto leftJacobianInverse(Eigen::Vector<Scalar, kNumParams> const&)
308  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
309  return Eigen::Matrix<Scalar, kPointDim, kPointDim>::Identity();
310  }
311

313  return std::vector<Eigen::Vector<Scalar, kDof>>({
314  Eigen::Vector<Scalar, kDof>({0.0}),
315  Eigen::Vector<Scalar, kDof>({0.00001}),
316  Eigen::Vector<Scalar, kDof>({1.0}),
317  Eigen::Vector<Scalar, kDof>({-1.0}),
318  Eigen::Vector<Scalar, kDof>({5.0}),

409 template <class TScalar, int kTranslationDim, template <class> class TLeftGroup>
410 requires LeftJacobianImplConcept<TLeftGroup<TScalar>>
411 class SemiDirectProductWithTranslation {
412  public:
413  using Scalar = TScalar;
414  using LeftGroup = TLeftGroup<Scalar>;
415 
416  // The is also the dimension of the translation.
417  static int const kPointDim = kTranslationDim;
418  static_assert(kPointDim == LeftGroup::kPointDim);
419 
420  static int const kDof = LeftGroup::kDof + kPointDim;
421  static int const kNumParams = LeftGroup::kNumParams + kPointDim;
422  static int const kAmbientDim = kPointDim + 1;
423 
424  static auto exp(Eigen::Vector<Scalar, kDof> tangent)
425  -> Eigen::Vector<Scalar, kNumParams> {
426  Eigen::Vector<Scalar, LeftGroup::kDof> left_params =
427  LeftGroup::exp(leftTangent(tangent));
428  return params(
429  left_params,
430  LeftGroup::leftJacobian(left_params) * translationTangent(tangent));
431  }
432 
433  static auto log(Eigen::Vector<Scalar, kNumParams> const& params)
434  -> Eigen::Vector<Scalar, kDof> {
435  Eigen::Vector<Scalar, LeftGroup::kNumParams> left_params =
436  leftParams(params);
437  return tangent(
438  LeftGroup::log(left_params),
439  LeftGroup::leftJacobianInverse(left_params) * rightParams(params));
440  }
441 
442  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& params)
443  -> Eigen::Vector<Scalar, kNumParams> {
444  Eigen::Vector<Scalar, LeftGroup::kNumParams> left_params =
445  LeftGroup::inverse(leftParams(params));
446  return params(
447  left_params, -LeftGroup::action(left_params, translation(params)));
448  }
449 
450  static auto action(
451  Eigen::Vector<Scalar, kNumParams> const& params,
452  Eigen::Vector<Scalar, kPointDim> const& point)
453  -> Eigen::Vector<Scalar, kPointDim> {
454  return LeftGroup::action(leftParams(params), point) + translation(params);
455  }
456 
457  static auto multiplication(
458  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
459  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
460  -> Eigen::Vector<Scalar, kNumParams> {
461  Eigen::Vector<Scalar, LeftGroup::kNumParams> left_params =
462  LeftGroup::multiplication(
463  leftParams(lhs_params), leftParams(rhs_params));
464  params(left_params, LeftGroup::action(left_params, translation(params)));
465  }
466 
467  static auto matrix(Eigen::Vector<Scalar, kNumParams> const& params)
468  -> Eigen::Matrix<Scalar, kPointDim, kAmbientDim> {
469  Eigen::Matrix<Scalar, kPointDim, kAmbientDim> mat;
470  mat.template topLeftCorner<kPointDim, kPointDim>() =
471  LeftGroup::matrix(leftParams(params));
472  mat.template topRightCorner<kPointDim, 1>() = translation(params);
473  }
474 
475  static auto homogeneousMatrix(Eigen::Vector<Scalar, kNumParams> const& params)
476  -> Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> {
477  Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> mat;
478  mat.template topLeftCorner<kPointDim, kAmbientDim>() = matrix(params);
479  mat.template bottomLeftCorner<1, kPointDim>().setZero();
480  mat(kPointDim, kPointDim) = 1.0;
481  }
482

539  -> Eigen::Vector<Scalar, kDof> {
