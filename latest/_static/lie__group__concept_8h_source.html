19 template <class T>
20 concept LieGroupImplConcept =
21  (T::kPointDim == 2 || T::kPointDim == 3) && // 2d or 3d points
22  (T::kPointDim == T::kAmbientDim // inhomogeneous point representation
23  ||
24  T::kPointDim + 1 == T::kAmbientDim) // or homogeneous point representation
25  && requires(
26  T g,
27  Eigen::Vector<typename T::Scalar, T::kDof> tangent,
28  Eigen::Vector<typename T::Scalar, T::kPointDim> point,
29  Eigen::Vector<typename T::Scalar, T::kNumParams> params) {
30  {
31  T::identityParams()
32  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kNumParams>>;
33 
34  {
35  T::exp(tangent)
36  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kNumParams>>;
37 
38  {
39  T::log(params)
40  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kDof>>;
41 
42  {
43  T::inverse(params)
44  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kNumParams>>;
45 
46  {
47  T::action(params, point)
48  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kPointDim>>;
49 
50  {
51  T::multiplication(params, params)
52  } -> ConvertibleTo<Eigen::Vector<typename T::Scalar, T::kNumParams>>;
53 
54  {
55  T::matrix(params)
56  } -> ConvertibleTo<
57  Eigen::Matrix<typename T::Scalar, T::kPointDim, T::kAmbientDim>>;
58 
59  {
60  T::homogeneousMatrix(params)
61  } -> ConvertibleTo<
62  Eigen::Matrix<typename T::Scalar, T::kAmbientDim, T::kAmbientDim>>;
63 
64  {
65  T::exampleTangents()
66  } -> ConvertibleTo<std::vector<Eigen::Vector<typename T::Scalar, T::kDof>>>;
67 
68  {
69  T::exampleParams()
70  } -> ConvertibleTo<
71  std::vector<Eigen::Vector<typename T::Scalar, T::kNumParams>>>;
72 
73  { T::areParamsValid(params) } -> ConvertibleTo<sophus::Expected<void>>;
74 };

90 template <
91  class TScalar,
92  template <class>
93  class TLeftGroup,
94  template <class>
95  class TRightGroup>
96 requires LieGroupImplConcept<TLeftGroup<TScalar>> &&
97  LieGroupImplConcept<TRightGroup<TScalar>>
98 class DirectProduct {
99  public:
100  using Scalar = TScalar;
101  using LeftGroup = TLeftGroup<Scalar>;
102  using RightGroup = TRightGroup<Scalar>;
103 
104  static int const kDof = LeftGroup::kDof + RightGroup::kDof;
105  static int const kNumParams = LeftGroup::kNumParams + RightGroup::kNumParams;
106  static int const kPointDim = LeftGroup::kPointDim;
107  static_assert(kPointDim == RightGroup::kPointDim);
108  static int const kAmbientDim = LeftGroup::kAmbientDim;
109  static_assert(kAmbientDim == RightGroup::kAmbientDim);
110  static_assert(kPointDim == kAmbientDim);
111 
112  static auto identityParams() -> Eigen::Vector<Scalar, kNumParams> {
113  return params(LeftGroup::identityParams(), RightGroup::identityParams());
114  }
115 
116  static auto exp(Eigen::Vector<Scalar, kDof> tangent)
117  -> Eigen::Vector<Scalar, kNumParams> {
118  return params(
119  LeftGroup::exp(leftTangent(tangent)),
120  RightGroup::exp(rightTangent(tangent)));
121  }
122 
123  static auto log(Eigen::Vector<Scalar, kNumParams> const& params)
124  -> Eigen::Vector<Scalar, kDof> {
125  return tangent(
126  LeftGroup::log(leftParams(params)),
127  RightGroup::log(rightParams(params)));
128  }
129 
130  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& params)
131  -> Eigen::Vector<Scalar, kNumParams> {
132  return params(
133  LeftGroup::inverse(leftParams(params)),
134  RightGroup::inverse(rightParams(params)));
135  }
136 
137  static auto multiplication(
138  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
139  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
140  -> Eigen::Vector<Scalar, kNumParams> {
141  return params(
142  LeftGroup::multiplication(
143  leftParams(lhs_params), leftParams(rhs_params)),
144  RightGroup::multiplication(
145  rightParams(lhs_params), rightParams(rhs_params)));
146  }
147 
148  static auto action(
149  Eigen::Vector<Scalar, kNumParams> const& params,
150  Eigen::Vector<Scalar, kPointDim> const& point)
151  -> Eigen::Vector<Scalar, kPointDim> {
152  return LeftGroup::action(RightGroup::action(point));
153  }
154 
155  static auto matrix(Eigen::Vector<Scalar, kNumParams> const& params)
156  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
157  return LeftGroup::matrix(leftParams(params)) *
158  RightGroup::matrix(rightParams(params));
159  }
160 
161  static auto homogeneousMatrix(Eigen::Vector<Scalar, kNumParams> const& params)
162  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
163  return matrix(params);
164  }
165 
166  static auto leftJacobian(Eigen::Vector<Scalar, kNumParams> const&)
167  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
168  return LeftGroup::leftJacobian(leftParams(params)) *
169  RightGroup::leftJacobian(rightParams(params));
170  }
171 
172  static auto leftJacobianInverse(Eigen::Vector<Scalar, kNumParams> const&)
173  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
174  return LeftGroup::leftJacobianInverse(leftParams(params)) *
175  RightGroup::leftJacobianInverse(rightParams(params));
176  }
177 
178  static auto exampleTangents() -> std::vector<Eigen::Vector<Scalar, kDof>> {
179  std::vector<Eigen::Vector<Scalar, kDof>> examples;
180  for (auto const& left_tangent : LeftGroup::exampleTangents()) {
181  for (auto const& right_tangents : LeftGroup::exampleTangents()) {
182  examples.push_back(tagent(left_tangent, right_tangents));
183  }
184  }
185  return examples;
186  }
187 
188  static auto exampleParams()
189  -> std::vector<Eigen::Vector<Scalar, kNumParams>> {
190  std::vector<Eigen::Vector<Scalar, kNumParams>> examples;
191  for (auto const& left_params : LeftGroup::exampleParams()) {
192  for (auto const& right_params : LeftGroup::exampleParams()) {
193  examples.push_back(params(left_params, right_params));
194  }
195  }
196  return examples;
197  }

199  static auto areParamsValid(Eigen::Vector<Scalar, kNumParams> const& params)
200  -> sophus::Expected<void> {
201  farm_ng::Error error;
202 
203  auto are_left_params_valid = LeftGroup::areParamsValid(leftParams(params));
204  auto are_right_params_valid =
205  LeftGroup::areParamsValid(rightParams(params));
206 
207  if (!are_left_params_valid) {
208  error.details.insert(
209  error.details.end(),
210  are_left_params_valid.error().details.begin(),
211  are_left_params_valid.error().details.end());
212  }
213  if (!are_right_params_valid) {
214  error.details.insert(
215  error.details.end(),
216  are_right_params_valid.error().details.begin(),
217  are_right_params_valid.error().details.end());
218  }
219 
220  if (!error.details.empty()) {
221  return tl::unexpected(error);
222  }
223  return sophus::Expected<void>{};
224  }

226  private:
227  static auto leftParams(Eigen::Vector<Scalar, kNumParams> const& params)
228  -> Eigen::Vector<Scalar, LeftGroup::kNumParams> {
229  return params.template head<LeftGroup::kNumParams>();
230  }
231 
232  static auto rightParams(Eigen::Vector<Scalar, kNumParams> const& params)
233  -> Eigen::Vector<Scalar, RightGroup::kNumParams> {
234  return params.template tail<RightGroup::kNumParams>();
235  }
236 
237  static auto params(
238  Eigen::Vector<Scalar, LeftGroup::kNumParams> const& params1,
239  Eigen::Vector<Scalar, RightGroup::kNumParams> const& params2)
240  -> Eigen::Vector<Scalar, kNumParams> {
241  Eigen::Vector<Scalar, kNumParams> params;
242  params.template head<LeftGroup::kNumParams>() = params1;
243  params.template tail<RightGroup::kNumParams>() = params2;
244  return params;
245  }
246 
247  static auto leftTangent(Eigen::Vector<Scalar, kDof> const& tangent)
248  -> Eigen::Vector<Scalar, LeftGroup::kDof> {
249  return tangent.template head<LeftGroup::kDof>();
250  }
251 
252  static auto rightTangent(Eigen::Vector<Scalar, kDof> const& tangent)
253  -> Eigen::Vector<Scalar, RightGroup::kDof> {
254  return tangent.template tail<RightGroup::kDof>();
255  }
256 
257  static auto tangent(
258  Eigen::Vector<Scalar, LeftGroup::kDof> const& tangent1,
259  Eigen::Vector<Scalar, RightGroup::kDof> const& tangent2)
260  -> Eigen::Vector<Scalar, kDof> {
261  Eigen::Vector<Scalar, kDof> tangent;
262  tangent.template head<LeftGroup::kDof>() = tangent1;
263  tangent.template tail<RightGroup::kDof>() = tangent2;
264  return tangent;
265  }

269 class Rotation2Impl {
270  public:
271  using Scalar = TScalar;
272  static int const kDof = 1;
273  static int const kNumParams = 2;
274  static int const kPointDim = 2;
275  static int const kAmbientDim = 2;
276 
277  static auto identityParams() -> Eigen::Vector<Scalar, kNumParams> {
278  return Eigen::Vector<Scalar, 2>(1.0, 0.0);
279  }
280 
281  static auto exp(Eigen::Vector<Scalar, kDof> const& angle)
282  -> Eigen::Vector<Scalar, kNumParams> {
283  using std::cos;
284  using std::sin;
285  return Eigen::Vector<Scalar, 2>(cos(angle[0]), sin(angle[0]));
286  }
287 
288  static auto log(Eigen::Vector<Scalar, kNumParams> const& unit_complex)
289  -> Eigen::Vector<Scalar, kDof> {
290  using std::atan2;
291  return Eigen::Vector<Scalar, 1>{atan2(unit_complex.y(), unit_complex.x())};
292  }
293 
294  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& unit_complex)
295  -> Eigen::Vector<Scalar, kNumParams> {
296  return Eigen::Vector<Scalar, kNumParams>(
297  unit_complex.x(), -unit_complex.y());
298  }
299 
300  static auto action(
301  Eigen::Vector<Scalar, kNumParams> const& unit_complex,
302  Eigen::Vector<Scalar, kPointDim> const& point)
303  -> Eigen::Vector<Scalar, kPointDim>;
304 
305  static auto multiplication(
306  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
307  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
308  -> Eigen::Vector<Scalar, kNumParams> {
309  Scalar const lhs_real = lhs_params.x();
310  Scalar const lhs_imag = lhs_params.y();
311  Scalar const rhs_real = rhs_params.x();
312  Scalar const rhs_imag = rhs_params.y();
313 
314  // complex multiplication
315  Scalar const result_real = lhs_real * rhs_real - lhs_imag * rhs_imag;
316  Scalar const result_imag = lhs_real * rhs_imag + lhs_imag * rhs_real;
317 
318  Scalar const squared_norm =
319  result_real * result_real + result_imag * result_imag;
320  // We can assume that the squared-norm is close to 1 since we deal with a
321  // unit complex number. Due to numerical precision issues, there might
322  // be a small drift after pose concatenation. Hence, we need to renormalizes
323  // the complex number here.
324  // Since squared-norm is close to 1, we do not need to calculate the costly
325  // square-root, but can use an approximation around 1 (see
326  // http://stackoverflow.com/a/12934750 for details).
327  if (squared_norm != 1.0) {
328  Scalar const scale = 2.0 / (1.0 + squared_norm);
329  return Eigen::Vector<Scalar, kNumParams>(
330  result_real * scale, result_imag * scale);
331  }
332  return Eigen::Vector<Scalar, kNumParams>(result_real, result_imag);
333  }
334 
335  static auto matrix(Eigen::Vector<Scalar, kNumParams> const& unit_complex)
336  -> Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim>;
337 
338  static auto homogeneousMatrix(
339  Eigen::Vector<Scalar, kNumParams> const& unit_complex)
340  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
341  return matrix(unit_complex);
342  }
343 
344  static auto leftJacobian(Eigen::Vector<Scalar, kNumParams> const&)
345  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
346  return Eigen::Matrix<Scalar, kPointDim, kPointDim>::Identity();
347  }
348 
349  static auto leftJacobianInverse(Eigen::Vector<Scalar, kNumParams> const&)
350  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
351  return Eigen::Matrix<Scalar, kPointDim, kPointDim>::Identity();
352  }

376  static auto areParamsValid(
377  Eigen::Vector<Scalar, kNumParams> const& unit_complex)
378  -> sophus::Expected<void> {
379  static const Scalar kThr = kEpsilon<Scalar> * kEpsilon<Scalar>;
380  const Scalar squared_norm = unit_complex.squaredNorm();
381 
382  if (!(squared_norm <= kThr)) {

386  unit_complex[0],
387  unit_complex[1],
388  squared_norm,
389  kThr);
390  }
391  return sophus::Expected<void>{};
392  }

497 template <class TScalar, int kTranslationDim, template <class> class TLeftGroup>
498 requires LeftJacobianImplConcept<TLeftGroup<TScalar>>
499 class SemiDirectProductWithTranslation {
500  public:
501  using Scalar = TScalar;
502  using LeftGroup = TLeftGroup<Scalar>;
503 
504  // The is also the dimension of the translation.
505  static int const kPointDim = kTranslationDim;
506  static_assert(kPointDim == LeftGroup::kPointDim);
507 
508  static int const kDof = LeftGroup::kDof + kPointDim;
509  static int const kNumParams = LeftGroup::kNumParams + kPointDim;
510  static int const kAmbientDim = kPointDim + 1;
511 
512  static auto identityParams() -> Eigen::Vector<Scalar, kNumParams> {
513  return params(
514  LeftGroup::identityParams(), Eigen::Vector<Scalar, kPointDim>::Zero());
515  }
516 
517  static auto exp(Eigen::Vector<Scalar, kDof> tangent)
518  -> Eigen::Vector<Scalar, kNumParams> {
519  Eigen::Vector<Scalar, LeftGroup::kNumParams> left_params =
520  LeftGroup::exp(leftTangent(tangent));
521  return params(
522  left_params,
523  LeftGroup::leftJacobian(left_params) * translationTangent(tangent));
524  }
525 
526  static auto log(Eigen::Vector<Scalar, kNumParams> const& params)
527  -> Eigen::Vector<Scalar, kDof> {
528  Eigen::Vector<Scalar, LeftGroup::kNumParams> left_params =
529  leftParams(params);
530  return tangent(
531  LeftGroup::log(left_params),
532  LeftGroup::leftJacobianInverse(left_params) * translation(params));
533  }
534 
535  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& params)
536  -> Eigen::Vector<Scalar, kNumParams> {
537  Eigen::Vector<Scalar, LeftGroup::kNumParams> left_params =
538  LeftGroup::inverse(leftParams(params));
539  return params(
540  left_params, -LeftGroup::action(left_params, translation(params)));
541  }
542 
543  static auto action(
544  Eigen::Vector<Scalar, kNumParams> const& params,
545  Eigen::Vector<Scalar, kPointDim> const& point)
546  -> Eigen::Vector<Scalar, kPointDim> {
547  return LeftGroup::action(leftParams(params), point) + translation(params);
548  }
549 
550  static auto multiplication(
551  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
552  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
553  -> Eigen::Vector<Scalar, kNumParams> {
554  Eigen::Vector<Scalar, LeftGroup::kNumParams> left_params =
555  LeftGroup::multiplication(
556  leftParams(lhs_params), leftParams(rhs_params));
557  params(left_params, LeftGroup::action(left_params, translation(params)));
558  }
559 
560  static auto matrix(Eigen::Vector<Scalar, kNumParams> const& params)
561  -> Eigen::Matrix<Scalar, kPointDim, kAmbientDim> {
562  Eigen::Matrix<Scalar, kPointDim, kAmbientDim> mat;
563  mat.template topLeftCorner<kPointDim, kPointDim>() =
564  LeftGroup::matrix(leftParams(params));
565  mat.template topRightCorner<kPointDim, 1>() = translation(params);
566  return mat;
567  }
568 
569  static auto homogeneousMatrix(Eigen::Vector<Scalar, kNumParams> const& params)
570  -> Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> {
571  Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> mat;
572  mat.template topLeftCorner<kPointDim, kAmbientDim>() = matrix(params);
573  mat.template bottomLeftCorner<1, kPointDim>().setZero();
574  mat(kPointDim, kPointDim) = 1.0;
575  }
