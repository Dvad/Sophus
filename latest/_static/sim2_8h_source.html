57 /// Sim2 base type - implements Sim2 class but is storage agnostic.
58 ///
59 /// Sim(2) is the group of rotations and translation and scaling in 2d. It is
60 /// the semi-direct product of R+xSO(2) and the 2d Euclidean vector space. The
61 /// class is represented using a composition of RxSo2 for scaling plus
62 /// rotation and a 2-vector for translation.
63 ///
64 /// - 3x3 Eigen::Matrix representation:
65 ///
66 /// ```
67 /// | s*R t |
68 /// | o 1 |
69 /// ```
70 ///
71 /// where ``R`` is a 3x3 rotation matrix, ``s`` a positive scale factor,
72 /// ``t`` a translation 3-vector and ``o`` a 3-column vector of zeros.
73 ///
74 /// - Tangent 4-vector: [upsilon, omega, sigma],
75 ///
76 /// where ``upsilon`` is the translational velocity 3-vector and ``omega`` the
77 /// rotational velocity 3-vector, and ``sigma = log(s)``.
78 ///
79 /// - Internal 4-vector representation: [t0, t1, zr, zi],
80 ///
81 /// with ``t0, t1`` are the translational components, and ``zr``
82 /// the real part and ``zi`` the imaginary part of a non-zero
83 /// complex number. Here the scale ``s`` is equal to the norm of the complex
84 /// number ``s = |z|``.
85 ///
86 /// See RxSo2 for more details of the scaling + rotation representation in 2d.
87 ///
88 ///
89 /// Sim(2) is neither compact, nor a commutative group.
90 ///
91 /// See RxSo2 for more details of the scaling + rotation representation in 2d.
92 ///
93 template <class TDerived>
94 class Sim2Base {
95  public:
96  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
97  using TranslationType =
98  typename Eigen::internal::traits<TDerived>::TranslationType;
99  using RxSo2Type = typename Eigen::internal::traits<TDerived>::RxSo2Type;

101  /// Degrees of freedom of manifold, number of dimensions in tangent space
102  /// (two for translation, one for rotation and one for scaling).
103  static int constexpr kDoF = 4;
104  /// Number of internal parameters used (2-tuple for complex number, two for
105  /// translation).
106  static int constexpr kNumParameters = 4;
107  /// Group transformations are 3x3 matrices.
108  static int constexpr kMatrixDim = 3;
109  /// Points are 2-dimensional
110  static int constexpr kPointDim = 2;
111  using Transformation = Eigen::Matrix<Scalar, kMatrixDim, kMatrixDim>;
112  using Point = Eigen::Vector2<Scalar>;
113  using HomogeneousPoint = Eigen::Vector3<Scalar>;
114  using Line = Eigen::ParametrizedLine<Scalar, 2>;
115  using Hyperplane = Eigen::Hyperplane<Scalar, 2>;
116  using Tangent = Eigen::Vector<Scalar, kDoF>;
117  using Adjoint = Eigen::Matrix<Scalar, kDoF, kDoF>;

136  /// Adjoint transformation
137  ///
138  /// This function return the adjoint transformation ``Ad`` of the group
139  /// element ``A`` such that for all ``x`` it holds that
140  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
141  ///
142  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
143  Adjoint res;
144  res.setZero();
145  res.template block<2, 2>(0, 0) = rxso2().matrix();
146  res(0, 2) = translation()[1];
147  res(1, 2) = -translation()[0];
148  res.template block<2, 1>(0, 3) = -translation();
149 
150  res(2, 2) = Scalar(1);
151 
152  res(3, 3) = Scalar(1);
153  return res;
154  }

165  /// Returns group inverse.
166  ///
167  SOPHUS_FUNC [[nodiscard]] Sim2<Scalar> inverse() const {
168  RxSo2<Scalar> inv_r = rxso2().inverse();
169  return Sim2<Scalar>(inv_r, inv_r * (translation() * Scalar(-1)));

172  /// Logarithmic map
173  ///
174  /// Computes the logarithm, the inverse of the group exponential which maps
175  /// element of the group (rigid body transformations) to elements of the
176  /// tangent space (twist).
177  ///
178  /// To be specific, this function computes ``vee(logmat(.))`` with
179  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
180  /// of Sim(2).
181  ///
182  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
183  /// The derivation of the closed-form Sim(2) logarithm for is done
184  /// analogously to the closed-form solution of the SE(2) logarithm, see
185  /// J. Gallier, D. Xu, "Computing exponentials of skew symmetric matrices
186  /// and logarithms of orthogonal matrices", IJRA 2002.
187  /// https:///pdfs.semanticscholar.org/cfe3/e4b39de63c8cabd89bf3feff7f5449fc981d.pdf
188  /// (Sec. 6., pp. 8)
189  Tangent res;
190  Eigen::Vector2<Scalar> const theta_sigma = rxso2().log();
191  Scalar const theta = theta_sigma[0];
192  Scalar const sigma = theta_sigma[1];
193  Eigen::Matrix2<Scalar> const omega = So2<Scalar>::hat(theta);
194  Eigen::Matrix2<Scalar> const w_inv =
195  details::calcWInv<Scalar, 2>(omega, theta, sigma, scale());
196 
197  res.segment(0, 2) = w_inv * translation();
198  res[2] = theta;
199  res[3] = sigma;
200  return res;
201  }

203  /// Returns 3x3 matrix representation of the instance.
204  ///
205  /// It has the following form:
206  ///
207  /// | s*R t |
208  /// | o 1 |
209  ///
210  /// where ``R`` is a 2x2 rotation matrix, ``s`` a scale factor, ``t`` a
211  /// translation 2-vector and ``o`` a 2-column vector of zeros.
212  ///
213  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
214  Transformation homogenious_matrix;
215  homogenious_matrix.template topLeftCorner<2, 3>() = matrix2x3();
216  homogenious_matrix.row(2) =
217  Eigen::Matrix<Scalar, 3, 1>(Scalar(0), Scalar(0), Scalar(1));
218  return homogenious_matrix;
219  }

221  /// Returns the significant first two rows of the matrix above.
222  ///
223  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, 2, 3> matrix2x3() const {
224  Eigen::Matrix<Scalar, 2, 3> matrix;
225  matrix.template topLeftCorner<2, 2>() = rxso2().matrix();
226  matrix.col(2) = translation();
227  return matrix;
228  }

240  /// Group multiplication, which is rotation plus scaling concatenation.
241  ///
242  /// Note: That scaling is calculated with saturation. See RxSo2 for
243  /// details.
244  ///
245  template <class TOtherDerived>
246  SOPHUS_FUNC Sim2Product<TOtherDerived> operator*(
247  Sim2Base<TOtherDerived> const& other) const {
248  return Sim2Product<TOtherDerived>(
249  rxso2() * other.rxso2(), translation() + rxso2() * other.translation());

252  /// Group action on 2-points.
253  ///
254  /// This function rotates, scales and translates a two dimensional point
255  /// ``p`` by the Sim(2) element ``(bar_sR_foo, t_bar)`` (= similarity
256  /// transformation):
257  ///
258  /// ``p_bar = bar_sR_foo * p_foo + t_bar``.
259  ///
260  template <
261  typename TPointDerived,
262  typename = typename std::enable_if<
263  IsFixedSizeVector<TPointDerived, 2>::value>::type>
264  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
265  Eigen::MatrixBase<TPointDerived> const& p) const {
266  return rxso2() * p + translation();
267  }
268 
269  /// Group action on homogeneous 2-points. See above for more details.
270  ///
271  template <
272  typename THPointDerived,
273  typename = typename std::enable_if<
274  IsFixedSizeVector<THPointDerived, 3>::value>::type>
275  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
276  Eigen::MatrixBase<THPointDerived> const& p) const {
277  PointProduct<THPointDerived> const tp =
278  rxso2() * p.template head<2>() + p(2) * translation();
279  return HomogeneousPointProduct<THPointDerived>(tp(0), tp(1), p(2));
280  }
281 
282  /// Group action on lines.
283  ///
284  /// This function rotates, scales and translates a parametrized line
285  /// ``l(t) = o + t * d`` by the Sim(2) element:
286  ///
287  /// Origin ``o`` is rotated, scaled and translated
288  /// Direction ``d`` is rotated
289  ///
290  SOPHUS_FUNC Line operator*(Line const& l) const {
291  Line rotated_line = rxso2() * l;
292  return Line(
293  rotated_line.origin() + translation(), rotated_line.direction());
294  }
295 
296  /// Group action on hyper-planes.
297  ///
298  /// This function rotates a hyper-plane ``n.x + d = 0`` by the Sim2
299  /// element:
300  ///
301  /// Normal vector ``n`` is rotated
302  /// Offset ``d`` is scaled and adjusted for translation
303  ///
304  /// Note that in 2d-case hyper-planes are just another parametrization of
305  /// lines
306  ///
307  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
308  Hyperplane const rotated = rxso2() * p;
309  return Hyperplane(
310  rotated.normal(),
311  rotated.offset() - translation().dot(rotated.normal()));
312  }

314  /// Returns internal parameters of Sim(2).
315  ///
316  /// It returns (c[0], c[1], t[0], t[1]),
317  /// with c being the complex number, t the translation 3-vector.
318  ///
319  SOPHUS_FUNC [[nodiscard]] Eigen::Vector<Scalar, kNumParameters> params()
320  const {
321  Eigen::Vector<Scalar, kNumParameters> p;

339  /// Returns derivative of this * Sim2::exp(x) wrt. x at x=0.
340  ///
341  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>
342  dxThisMulExpXAt0() const {
343  Eigen::Matrix<Scalar, kNumParameters, kDoF> d;
344  d.template block<2, 2>(0, 0).setZero();
345  d.template block<2, 2>(0, 2) = rxso2().dxThisMulExpXAt0();
346  d.template block<2, 2>(2, 2).setZero();
347  d.template block<2, 2>(2, 0) = rxso2().matrix();
348  return d;
349  }

351  /// Returns derivative of log(this^{-1} * x) by x at x=this.
352  ///
353  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>
354  dxLogThisInvTimesXAtThis() const {
355  Eigen::Matrix<Scalar, kNumParameters, kDoF> d;
356  d.template block<2, 2>(0, 0).setZero();
357  d.template block<2, 2>(0, 2) = rxso2().inverse().matrix();
358  d.template block<2, 2>(2, 0) = rxso2().dxLogThisInvTimesXAtThis();
359  d.template block<2, 2>(2, 2).setZero();
360  return d;
361  }

554  /// Returns derivative of exp(x) wrt. x_i at x=0.
555  ///
556  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpXAt0() {
557  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
558  j.template block<2, 2>(0, 0).setZero();
559  j.template block<2, 2>(0, 2) = RxSo2<Scalar>::dxExpXAt0();
560  j.template block<2, 2>(2, 0).setIdentity();
561  j.template block<2, 2>(2, 2).setZero();
562  return j;

565  /// Returns derivative of exp(x) wrt. x.
566  ///
567  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
568  Tangent const& vec_a) {
569  static Eigen::Matrix2<Scalar> const kI = Eigen::Matrix2<Scalar>::Identity();
570  static Scalar const kOne(1.0);
571 
572  Scalar const theta = vec_a[2];
573  Scalar const sigma = vec_a[3];
574 
575  Eigen::Matrix2<Scalar> const omega = So2<Scalar>::hat(theta);
576  Eigen::Matrix2<Scalar> const omega_dtheta = So2<Scalar>::hat(kOne);
577  Eigen::Matrix2<Scalar> const omega2 = omega * omega;
578  Eigen::Matrix2<Scalar> const omega2_dtheta =
579  omega_dtheta * omega + omega * omega_dtheta;
580  Eigen::Matrix2<Scalar> const w =
581  details::calcW<Scalar, 2>(omega, theta, sigma);
582  Eigen::Vector2<Scalar> const upsilon = vec_a.segment(0, 2);
583 
584  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
585  j.template block<2, 2>(0, 0).setZero();
586  j.template block<2, 2>(0, 2) =
587  RxSo2<Scalar>::dxExpX(vec_a.template tail<2>());
588  j.template block<2, 2>(2, 0) = w;
589 
590  Scalar a;
591  Scalar b;
592  Scalar c;
593  Scalar a_dtheta;
594  Scalar b_dtheta;
595  Scalar a_dsigma;
596  Scalar b_dsigma;
597  Scalar c_dsigma;
598  details::calcWDerivatives(
599  theta,
600  sigma,
601  a,
602  b,
603  c,
604  a_dsigma,
605  b_dsigma,
606  c_dsigma,
607  a_dtheta,
608  b_dtheta);
609 
610  j.template block<2, 1>(2, 2) = (a_dtheta * omega + a * omega_dtheta +
611  b_dtheta * omega2 + b * omega2_dtheta) *
612  upsilon;
613  j.template block<2, 1>(2, 3) =
614  (a_dsigma * omega + b_dsigma * omega2 + c_dsigma * kI) * upsilon;
615 
616  return j;
617  }

633  return generator(i);

636  /// Derivative of Lie bracket with respect to first element.
637  ///
638  /// This function returns ``D_a [a, b]`` with ``D_a`` being the
639  /// differential operator with respect to ``a``, ``[a, b]`` being the lie
640  /// bracket of the Lie algebra sim(2).
641  /// See ``lieBracket()`` below.
642  ///
643 
644  /// Group exponential
645  ///
646  /// This functions takes in an element of tangent space and returns the
647  /// corresponding element of the group Sim(2).
648  ///
649  /// The first two components of ``a`` represent the translational part
650  /// ``upsilon`` in the tangent space of Sim(2), the following two components
651  /// of ``a`` represents the rotation ``theta`` and the final component
652  /// represents the logarithm of the scaling factor ``sigma``.
653  /// To be more specific, this function computes ``expmat(hat(a))`` with
654  /// ``expmat(.)`` being the matrix exponential and ``hat(.)`` the hat-operator
655  /// of Sim(2), see below.
656  ///
657  SOPHUS_FUNC static Sim2<Scalar> exp(Tangent const& vec_a) {
658  // For the derivation of the exponential map of Sim(kMatrixDim) see
659  // H. Strasdat, "Local Accuracy and Global Consistency for Efficient Visual
660  // SLAM", PhD thesis, 2012.
661  // http:///hauke.strasdat.net/files/strasdat_thesis_2012.pdf (A.5, pp. 186)
662  Eigen::Vector2<Scalar> const upsilon = vec_a.segment(0, 2);
663  Scalar const theta = vec_a[2];
664  Scalar const sigma = vec_a[3];
665  RxSo2<Scalar> rxso2 = RxSo2<Scalar>::exp(vec_a.template tail<2>());
666  Eigen::Matrix2<Scalar> const omega = So2<Scalar>::hat(theta);
667  Eigen::Matrix2<Scalar> const w =
668  details::calcW<Scalar, 2>(omega, theta, sigma);
669  return Sim2<Scalar>(rxso2, w * upsilon);
670  }
671 
672  /// Returns the ith infinitesimal generators of Sim(2).
673  ///
674  /// The infinitesimal generators of Sim(2) are:
675  ///
676  /// ```
677  /// | 0 0 1 |
678  /// G_0 = | 0 0 0 |
679  /// | 0 0 0 |
680  ///
681  /// | 0 0 0 |
682  /// G_1 = | 0 0 1 |
683  /// | 0 0 0 |
684  ///
685  /// | 0 -1 0 |
686  /// G_2 = | 1 0 0 |
687  /// | 0 0 0 |
688  ///
689  /// | 1 0 0 |
690  /// G_3 = | 0 1 0 |
691  /// | 0 0 0 |
692  /// ```
693  ///
694  /// Precondition: ``i`` must be in [0, 3].
695  ///
696  SOPHUS_FUNC static Transformation generator(int i) {
697  SOPHUS_ASSERT(i >= 0 || i <= 3, "i should be in range [0,3].");
698  Tangent e;
699  e.setZero();
700  e[i] = Scalar(1);
701  return hat(e);
702  }
703 
704  /// hat-operator
705  ///
706  /// It takes in the 4-vector representation and returns the corresponding
707  /// matrix representation of Lie algebra element.
708  ///
709  /// Formally, the hat()-operator of Sim(2) is defined as
710  ///
711  /// ``hat(.): R^4 -> R^{3x3}, hat(a) = sum_i a_i * G_i`` (for i=0,...,6)
712  ///
713  /// with ``G_i`` being the ith infinitesimal generator of Sim(2).
714  ///
715  /// The corresponding inverse is the vee()-operator, see below.
716  ///
717  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
718  Transformation omega;
719  omega.template topLeftCorner<2, 2>() =
720  RxSo2<Scalar>::hat(vec_a.template tail<2>());
721  omega.col(2).template head<2>() = vec_a.template head<2>();
722  omega.row(2).setZero();
723  return omega;
724  }
725 
726  /// Lie bracket
727  ///
728  /// It computes the Lie bracket of Sim(2). To be more specific, it computes
729  ///
730  /// ``[omega_1, omega_2]_sim2 := vee([hat(omega_1), hat(omega_2)])``
731  ///
732  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
733  /// hat()-operator and ``vee(.)`` the vee()-operator of Sim(2).
734  ///
735  SOPHUS_FUNC static Tangent lieBracket(
736  Tangent const& vec_a, Tangent const& vec_b) {
737  Eigen::Vector2<Scalar> const vec_upsilon1 = vec_a.template head<2>();
738  Eigen::Vector2<Scalar> const vec_upsilon2 = vec_b.template head<2>();
739  Scalar const theta1 = vec_a[2];
740  Scalar const theta2 = vec_b[2];
741  Scalar const sigma1 = vec_a[3];
742  Scalar const sigma2 = vec_b[3];
743 
744  Tangent res;
745  res[0] = -theta1 * vec_upsilon2[1] + theta2 * vec_upsilon1[1] +
746  sigma1 * vec_upsilon2[0] - sigma2 * vec_upsilon1[0];
747  res[1] = theta1 * vec_upsilon2[0] - theta2 * vec_upsilon1[0] +
748  sigma1 * vec_upsilon2[1] - sigma2 * vec_upsilon1[1];
749  res[2] = Scalar(0);
750  res[3] = Scalar(0);
751 
752  return res;
753  }
754 
755  /// Draw uniform sample from Sim(2) manifold.
756  ///
757  /// Translations are drawn component-wise from the range [-1, 1].
758  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
759  /// hence the scale is in [0.5, 2].
760  ///
761  template <class TUniformRandomBitGenerator>
762  static Sim2 sampleUniform(TUniformRandomBitGenerator& generator) {
763  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
764  return Sim2(
765  RxSo2<Scalar>::sampleUniform(generator),
766  Eigen::Vector2<Scalar>(uniform(generator), uniform(generator)));
767  }
768 
769  /// vee-operator
770  ///
771  /// It takes the 3x3-matrix representation ``Omega`` and maps it to the
772  /// corresponding 4-vector representation of Lie algebra.
773  ///
774  /// This is the inverse of the hat()-operator, see above.
775  ///
776  /// Precondition: ``Omega`` must have the following structure:
777  ///
778  /// | d -c a |
779  /// | c d b |
780  /// | 0 0 0 |
781  ///
782  SOPHUS_FUNC static Tangent vee(Transformation const& mat_omega) {
783  Tangent upsilon_omega_sigma;
784  upsilon_omega_sigma.template head<2>() =
785  mat_omega.col(2).template head<2>();
786  upsilon_omega_sigma.template tail<2>() =
787  RxSo2<Scalar>::vee(mat_omega.template topLeftCorner<2, 2>());
788  return upsilon_omega_sigma;
789  }
790 
791  protected:
792  RxSo2Member rxso2_; // NOLINT
793  TranslationMember translation_; // NOLINT
794 };
