55 // base case
56 template <class TUserFunc, typename TRuntimeImage, typename TPixelType>
57 struct VisitImpl<TUserFunc, TRuntimeImage, std::variant<TPixelType>> {
58  static void visit(TUserFunc&& func, TRuntimeImage const& image) {
59  if (image.pixelType().template is<TPixelType>()) {
60  func(image.template imageView<TPixelType>());
61  }
62  }
63 };
64 
65 // inductive case
66 template <
67  typename TUserFunc,
68  typename TRuntimeImage,
69  typename TPixelType,
70  typename... TRest>
71 struct VisitImpl<TUserFunc, TRuntimeImage, std::variant<TPixelType, TRest...>> {
72  static void visit(TUserFunc&& func, TRuntimeImage const& image) {
73  if (image.pixelType().template is<TPixelType>()) {
74  func(image.template imageView<TPixelType>());
75  } else {
76  VisitImpl<TUserFunc, TRuntimeImage, std::variant<TRest...>>::visit(
77  std::forward<TUserFunc>(func), image);
78  }
79  }
80 };
