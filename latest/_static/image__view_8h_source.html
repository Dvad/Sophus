14 /// Note that it is a conscious API decision to follow "shallow-compare" type
15 /// semantic for ImageView, MutImageView, Image and MutImage. Similar
16 /// "shallow-compare" types are: std::span (shallow-compare reference type), and
17 /// std::unique_ptr (shallow compare unique ownership type).
18 /// This is in contrast to regular types such as std::vector, std::string and
19 /// reference types which mimic regular type semantic such as std::string_view.

37 /// A view of an (immutable) image, which does not own the data.
38 ///
39 /// The API of ImageView allows for read-only access. There is an escape hook
40 /// for write access, see MutImageView::unsafeConstCast below.
41 ///
42 /// ImageViews are nullable. In that case `this->isEmpty()` is true.
43 ///
44 ///
45 /// Details on equality comparison, the state of the object, and
46 /// const-correctness.
47 ///
48 /// ImageView is a "shallow-compare type" similar to std::span<Pixel const> and
49 /// std::unique_ptr<Pixel const>. In particular, we define that the state of an
50 /// ImageView instance consists of the shape of the image ``shape_`` (see
51 /// ImageShape) and the pointer address to the first pixel ``ptr_``. No
52 /// public member method can change the pointer nor the shape, hence they are
53 /// all marked const.
54 template <class TPixel>
55 struct ImageView {
56  using PixelType = TPixel;
57 
58  /// Default constructor creates an empty image.
59  ImageView() = default;
60 
61  /// Creates view from shape and pointer to first pixel.
62  ImageView(ImageShape shape, TPixel const* ptr) noexcept
63  : shape_(shape), ptr_(ptr) {}
64 
65  /// Creates view from image size and pointer to first pixel. The image is
66  /// assumed to be contiguous and the pitch is set accordingly.
67  ImageView(sophus::ImageSize image_size, TPixel const* ptr) noexcept
68  : ImageView(ImageShape::makeFromSize<TPixel>(image_size), ptr) {}
69 
70  /// Returns true if view is empty.
71  [[nodiscard]] bool isEmpty() const { return this->ptr_ == nullptr; }
72 
73  /// Returns true if view is contiguous.
74  [[nodiscard]] bool isContiguous() const {
75  return imageSize().width * sizeof(TPixel) == shape().pitchBytes();
76  }
77 
78  /// Returns ImageSize.
79  /// It is {0,0} if view is empty.
80  [[nodiscard]] sophus::ImageSize const& imageSize() const {
81  return shape_.imageSize();
82  }
83 
84  /// Returns ImageShape.
85  /// It is {{0,0}, 0} is view is empty.
86  [[nodiscard]] ImageShape const& shape() const { return shape_; }
87 
88  [[nodiscard]] int width() const { return shape().width(); }
89  [[nodiscard]] int height() const { return shape().height(); }
90  [[nodiscard]] size_t pitchBytes() const { return shape().pitchBytes(); }
91 
92  /// Returns true if u is in [0, width).
93  [[nodiscard]] bool colInBounds(int u) const {
94  return u >= 0 && u < shape_.width();
95  }
96 
97  /// Returns true if v is in [0, height).
98  [[nodiscard]] bool rowInBounds(int v) const {
99  return v >= 0 && v < shape_.height();
100  }
101 
102  /// Returns v-th row pointer.
103  ///
104  /// Precondition: v must be in [0, height).
105  [[nodiscard]] TPixel const* rowPtr(int v) const {
106  return (TPixel*)((uint8_t*)(ptr_) + v * shape_.pitchBytes());
107  }
108 
109  /// Returns pixel u, v.
110  ///
111  /// Precondition: u must be in [0, width) and v must be in [0, height).
112  ///
113  /// Note:
114  /// * No panic if u.v is invalid,
115  ///
116  /// This is not the most necessarily the efficient function to call - e.g.
117  /// when iterating over the whole image. Use the following instead:
118  ///
119  /// for (int v=0; v<view.shape().height(); ++v) {
120  /// TPixel const* row = img.rowPtr(v);
121  /// for (int u=0; u<view.shape().width(); ++u) {
122  /// PixetT p = row[u];
123  /// }
124  /// }
125  [[nodiscard]] TPixel const& operator()(int u, int v) const {
126  return rowPtr(v)[u];
127  }
128 
129  [[nodiscard]] TPixel const& operator()(Eigen::Vector2i uv) const {
130  return this->operator()(uv[0], uv[1]);
131  }
132 
133  /// Returns pointer to first pixel.
134  [[nodiscard]] TPixel const* ptr() const { return ptr_; }
135 
136  /// Returns subview.
137  [[nodiscard]] ImageView subview(
138  Eigen::Vector2i uv, sophus::ImageSize size) const {
139  SOPHUS_ASSERT(colInBounds(uv[0]));
140  SOPHUS_ASSERT(rowInBounds(uv[1]));
141  SOPHUS_ASSERT_LE(uv.x() + size.width, shape_.width());
142  SOPHUS_ASSERT_LE(uv.y() + size.height, shape_.height());
143  return ImageView(
144  ImageShape::makeFromSizeAndPitch<TPixel>(size, shape_.pitchBytes()),
145  rowPtr(uv.y()) + uv.x());
146  }
147 
148  /// Performs reduction / fold on image view.
149  template <class TFunc>
150  void visit(TFunc const& user_function) const {
151  SOPHUS_ASSERT(!this->isEmpty());
152 
153  for (int v = 0; v < this->shape_.height(); ++v) {
154  TPixel const* p = this->rowPtr(v);
155  TPixel const* end_of_row = p + this->shape_.width();
156  for (; p != end_of_row; ++p) {
157  user_function(*p);
158  }
159  }
160  }
161 
162  /// Performs reduction / fold on image view.
163  template <class TReduceOp, class TVal>
164  [[nodiscard]] TVal reduce(
165  TReduceOp const& reduce_op, TVal val = TVal{}) const {
166  SOPHUS_ASSERT(!this->isEmpty()); // NOLINT
167 
168  for (int v = 0; v < this->shape_.height(); ++v) {
169  TPixel const* p = this->rowPtr(v);
170  TPixel const* end_of_row = p + this->shape_.width();
171  for (; p != end_of_row; ++p) {
172  reduce_op(*p, val);
173  }
174  }
175  return val;
176  }
177 
178  /// Performs reduction / fold on image view with short circuit condition.
179  template <class TShortCircuitReduceOp, class TVal>
180  [[nodiscard]] TVal shortCircuitReduce(
181  TShortCircuitReduceOp const& short_circuit_reduce_op,
182  TVal val = TVal{}) const {
183  SOPHUS_ASSERT(!this->isEmpty());
184 
185  for (int v = 0; v < this->shape_.height(); ++v) {
186  TPixel const* p = this->rowPtr(v);
187  TPixel const* end_of_row = p + this->shape_.width();
188  for (; p != end_of_row; ++p) {
189  if (short_circuit_reduce_op(*p, val)) {
190  return val;
191  }
192  }
193  }
194  return val;
195  }
196 
197  /// The equality operator is deleted to avoid confusion. Since ImageView is a
198  /// "shallow-copy" type, a consistently defined equality would check for
199  /// equality of its (shallow) state:
200  ///
201  /// ```this->shape_ == rhs.shape() && this->ptr_ == rhs.ptr_````
202  ///
203  /// However, some users might expect that equality would check for pixel
204  /// values equality and return true for identical copies of data blocks.
205  ///
206  /// Here we follow std::span which also does not offer equality comparions.
207  bool operator==(ImageView const& rhs) const = delete;
208 
209  /// The in-equality operator is deleted to avoid confusion.
210  bool operator!=(ImageView const& rhs) const = delete;
211 
212  /// Returns true both views have the same size and contain the same data.
213  [[nodiscard]] bool hasSameData(ImageView const& rhs) const {
214  if (!(this->imageSize() == rhs.imageSize())) {
215  return false;
216  }
217  for (int v = 0; v < this->shape_.height(); ++v) {
218  TPixel const* p = this->rowPtr(v);
219  TPixel const* rhs_p = rhs.rowPtr(v);
220 
221  TPixel const* end_of_row = p + this->shape_.width();
222  for (; p != end_of_row; ++p, ++rhs_p) {
223  if (*p != *rhs_p) {
224  return false;
225  }
226  }
227  }
228  return true;
229  }
230 
231  protected:
232  /// Resets view such that it is empty.
233  void setViewToEmpty() { *this = {}; }
234 
235  ImageShape shape_ = {}; // NOLINT
236  TPixel const* ptr_ = nullptr; // NOLINT
237 
238  private:
239  template <class TT, template <class> class TAllocator>
240  friend class MutImage;
241 
242  template <class TT, template <class> class TAllocator>
243  friend class Image;
244 };

256 /// View of a mutable image, which does not own the data.
257 ///
258 /// The API of MutImageView allows for read and write access.
259 ///
260 /// MutImageView is nullable. In that case `this->isEmpty()` is true.
261 ///
262 ///
263 /// Details on equality comparison, the state of the object, and
264 /// const-correctness.
265 ///
266 /// MutImageView is a "shallow-compare type" similar to std::span<Pixel> and
267 /// std::unique_ptr<Pixel>. As ImageView, its state consists of the image shape
268 /// as well as the pointer address, and comparing those entities establishes
269 /// equality comparisons. Furthermore, giving mutable access to pixels is
270 /// considered a const operation, as in
271 ///
272 /// ```TPixel& mut(int u, int v) const```
273 ///
274 /// since this merely allows for changing a pixel value, but not its state
275 /// (data location and layout).
276 template <class TPixel>
277 class MutImageView : public ImageView<TPixel> {
278  public:
279  /// Default constructor creates an empty image.
280  MutImageView() = default;
281 
282  /// Creates view from shape and pointer to first pixel.
283  MutImageView(ImageShape shape, TPixel* ptr) noexcept
284  : ImageView<TPixel>(shape, ptr) {}
285 
286  /// Creates view from image size and pointer to first pixel. The image is
287  /// assumed to be contiguous and the pitch is set accordingly.
288  MutImageView(sophus::ImageSize image_size, TPixel* ptr) noexcept
289  : ImageView<TPixel>(image_size, ptr) {}
290 
291  /// Creates mutable view from view.
292  ///
293  /// It is the user's responsibility to make sure that the data owned by
294  /// the view can be modified safely.
295  [[nodiscard]] static MutImageView unsafeConstCast(ImageView<TPixel> view) {
296  return MutImageView(view.shape(), const_cast<TPixel*>(view.ptr()));
297  }
298 
299  /// Returns ImageView(*this).
300  ///
301  /// Returns non-mut version of view.
302  [[nodiscard]] ImageView<TPixel> view() const {
303  return ImageView<TPixel>(this->shape(), this->ptr());
304  }
305 
306  /// Copies data from view into this.
307  ///
308  /// Preconditions:
309  /// * this->isEmpty() == view.isEmpty()
310  /// * this->size() == view.size()
311  ///
312  /// No-op if view is empty.
313  void copyDataFrom(ImageView<TPixel> view) const {
314  SOPHUS_ASSERT_EQ(this->isEmpty(), view.isEmpty());
315 
316  if (this->isEmpty()) {
317  return;
318  }
319  SOPHUS_ASSERT_EQ(this->imageSize(), view.imageSize());
320  details::pitchedCopy(
321  (uint8_t*)this->ptr(),
322  this->shape().pitchBytes(),
323  (uint8_t const*)view.ptr(),
324  view.shape().pitchBytes(),
325  this->imageSize(),
326  sizeof(TPixel));
327  }
328 
329  /// Returns v-th row pointer of mutable pixel.
330  [[nodiscard]] TPixel* rowPtrMut(int v) const {
331  return (TPixel*)((uint8_t*)(this->ptr()) + v * this->shape_.pitchBytes());
332  }
333 
334  /// Mutable accessor to pixel u, v.
335  ///
336  /// Precondition: u must be in [0, width) and v must be in [0, height).
337  /// Silent UB on failure.
338  [[nodiscard]] TPixel& mut(int u, int v) const { return rowPtrMut(v)[u]; }
339 
340  [[nodiscard]] TPixel& mut(Eigen::Vector2i uv) const {
341  return mut(uv[0], uv[1]);
342  }
343 
344  /// Mutates each pixel of this with given unary operation
345  ///
346  /// Preconditions: this must not be empty.
347  template <class TUnaryOperation>
348  void mutate(TUnaryOperation const& unary_op) const {
349  SOPHUS_ASSERT(!this->isEmpty());
350 
351  for (int v = 0; v < this->shape_.height(); ++v) {
352  TPixel* p = this->rowPtrMut(v);
353  TPixel const* end_of_row = p + this->shape_.width();
354  for (; p != end_of_row; ++p) {
355  *p = unary_op(*p);
356  }
357  }
358  }
359 
360  /// For each pixel in `this` with coordinates (u,v), populates with the user
361  /// provided function, evaluated as `uv_op(u,v)`, where u and v are integers
362  /// such that u in [0, width), v in [0, height)
363  ///
364  /// Preconditions: this must not be empty.
365  template <class TUVOperation>
366  void generate(TUVOperation const& uv_op) const {
367  SOPHUS_ASSERT(!this->isEmpty());
368 
369  for (int v = 0; v < this->shape_.height(); ++v) {
370  TPixel* p = this->rowPtrMut(v);
371  TPixel const* end_of_row = p + this->shape_.width();
372  for (int u = 0; p != end_of_row; ++p, ++u) {
373  *p = uv_op(u, v);
374  }
375  }
376  }
377 
378  /// Transforms view using unary operation and assigns result to this.
379  ///
380  /// Preconditions:
381  /// - this must not be empty.
382  /// - this->imageSize() == view.imageSize()
383  template <class TOtherPixel, class TUnaryOperation>
384  void transformFrom(
385  ImageView<TOtherPixel> view, TUnaryOperation const& unary_op) const {
386  SOPHUS_ASSERT(!this->isEmpty());
387  SOPHUS_ASSERT_EQ(view.imageSize(), this->imageSize());
388 
389  for (int v = 0; v < this->shape_.height(); ++v) {
390  TPixel* mut_p = this->rowPtrMut(v);
391  TOtherPixel const* p = view.rowPtr(v);
392  TOtherPixel const* end_of_row = p + view.shape().width();
393 
394  for (; p != end_of_row; ++p, ++mut_p) {
395  *mut_p = unary_op(*p);
396  }
397  }
398  }
399 
400  /// Transforms two views using binary operation and assigns result to this.
401  ///
402  /// Preconditions:
403  /// - this must not be empty.
404  /// - this->imageSize() == lhs.imageSize() == rhs.imageSize()
405  template <class TLhsPixel, class TRhsPixel, class TBinaryOperation>
406  void transformFrom(
407  ImageView<TLhsPixel> lhs,
408  ImageView<TRhsPixel> rhs,
409  TBinaryOperation const& binary_op) const {
410  SOPHUS_ASSERT(!this->isEmpty());
411  SOPHUS_ASSERT_EQ(lhs.imageSize(), this->imageSize());
412  SOPHUS_ASSERT_EQ(rhs.imageSize(), this->imageSize());
413 
414  for (int v = 0; v < this->shape_.height(); ++v) {
415  TPixel* mut_p = this->rowPtrMut(v);
416  TLhsPixel const* lhs_p = lhs.rowPtr(v);
417  TRhsPixel const* rhs_p = rhs.rowPtr(v);
418 
419  for (int u = 0; u < this->shape_.width(); ++u) {
420  mut_p[u] = binary_op(lhs_p[u], rhs_p[u]);
421  }
422  }
423  }
424 
425  /// Populates every pixel of this with val.
426  ///
427  /// Preconditions: this must not be empty.
428  void fill(TPixel const& val) const {
429  mutate([&](TPixel const& /*unused*/) { return val; });
430  }
431 
432  /// Returns pointer of mutable data to first pixel.
433  [[nodiscard]] TPixel* ptrMut() const {
434  return const_cast<TPixel*>(ImageView<TPixel>::ptr());
435  }
436 
437  /// Returns mutable subview.
438  [[nodiscard]] MutImageView mutSubview(
439  Eigen::Vector2i uv, sophus::ImageSize size) const {
440  SOPHUS_ASSERT(this->colInBounds(uv[0]));
441  SOPHUS_ASSERT(this->rowInBounds(uv[1]));
442  SOPHUS_ASSERT_LE(uv.x() + size.width, this->shape().width());
443  SOPHUS_ASSERT_LE(uv.y() + size.height, this->shape().height());
444 
445  return MutImageView(
446  ImageShape::makeFromSizeAndPitch<TPixel>(
447  size, this->shape().pitchBytes()),
448  this->rowPtrMut(uv.y()) + uv.x());
449  }
450 };
