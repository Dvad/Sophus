60 /// Cartesian base type - implements Cartesian class but is storage agnostic.
61 ///
62 /// Euclidean vector space as Lie group.
63 ///
64 /// Lie groups can be seen as a generalization over the Euclidean vector
65 /// space R^M. Here a kMatrixDim-dimensional vector ``p`` is represented as a
66 // (M+1) x (M+1) homogeneous matrix:
67 ///
68 /// | I p |
69 /// | o 1 |
70 ///
71 /// On the other hand, Cartesian(M) can be seen as a special case of SE(M)
72 /// with identity rotation, and hence represents pure translation.
73 ///
74 /// The purpose of this class is two-fold:
75 /// - for educational purpose, to highlight how Lie groups generalize over
76 /// Euclidean vector spaces.
77 /// - to be used in templated/generic algorithms (such as sophus::Spline)
78 /// which are implemented against the Lie group interface.
79 ///
80 /// Obviously, Cartesian(M) can just be represented as a M-tuple.
81 ///
82 /// Cartesian is not compact, but a commutative group. For vector additions it
83 /// holds `a+b = b+a`.
84 ///
85 /// See Cartesian class for more details below.
86 ///
87 template <class TDerived, int kM>
88 class CartesianBase {
89  public:
90  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
91  using ParamsType = typename Eigen::internal::traits<TDerived>::ParamsType;
92  /// Degrees of freedom of manifold, equals to number of Cartesian coordinates.
93  static int constexpr kDoF = kM;
94  /// Number of internal parameters used, also M.
95  static int constexpr kNumParameters = kM;
96  /// Group transformations are (M+1)x(M+1) matrices.
97  static int constexpr kMatrixDim = kM + 1;
98  static int constexpr kPointDim = kM;
99 
100  using Transformation = Eigen::Matrix<Scalar, kMatrixDim, kMatrixDim>;
101  using Point = Eigen::Vector<Scalar, kM>;
102  using HomogeneousPoint = Eigen::Vector<Scalar, kMatrixDim>;
103  using Line = Eigen::ParametrizedLine<Scalar, kM>;
104  using Hyperplane = Eigen::Hyperplane<Scalar, kM>;
105  using Tangent = Eigen::Vector<Scalar, kDoF>;
106  using Adjoint = Eigen::Matrix<Scalar, kDoF, kDoF>;
107 
108  /// For binary operations the return type is determined with the
109  /// ScalarBinaryOpTraits feature of Eigen. This allows mixing concrete and Map
110  /// types, as well as other compatible scalar types such as Ceres::Jet and
111  /// double scalars with Cartesian operations.
112  template <class TOtherDerived>
113  using ReturnScalar = typename Eigen::
114  ScalarBinaryOpTraits<Scalar, typename TOtherDerived::Scalar>::ReturnType;
115 
116  template <class TOtherDerived>
117  using CartesianSum = Cartesian<ReturnScalar<TOtherDerived>, kM>;
118 
119  template <class TPointDerived>
120  using PointProduct = Eigen::Vector<ReturnScalar<TPointDerived>, kM>;
121 
122  template <class THPointDerived>
123  using HomogeneousPointProduct =
124  Eigen::Vector<ReturnScalar<THPointDerived>, kMatrixDim>;
125 
126  /// Adjoint transformation
127  ///
128  /// Always identity of commutative groups.
129  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const { return Adjoint::Identity(); }
130 
131  /// Returns copy of instance casted to NewScalarType.
132  ///
133  template <class TNewScalarType>
134  SOPHUS_FUNC Cartesian<TNewScalarType, kM> cast() const {
135  return Cartesian<TNewScalarType, kM>(
136  params().template cast<TNewScalarType>());
137  }

142  dxThisMulExpXAt0() const {
143  Eigen::Matrix<Scalar, kNumParameters, kDoF> m;
144  m.setIdentity();
145  return m;
146  }
147 
148  /// Returns derivative of log(this^{-1} * x) by x at x=this.
149  ///
150  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>
151  dxLogThisInvTimesXAtThis() const {
152  Eigen::Matrix<Scalar, kDoF, kNumParameters> m;
153  m.setIdentity();
154  return m;
155  }
156 
157  /// Returns group inverse.
158  ///
159  /// The additive inverse.
160  ///
161  SOPHUS_FUNC [[nodiscard]] Cartesian<Scalar, kM> inverse() const {
162  return Cartesian<Scalar, kM>(-params());
163  }
164 
165  /// Logarithmic map
166  ///
167  /// For Euclidean vector space, just the identity. Or to be more precise
168  /// it just extracts the significant M-vector from the NxN matrix.
169  ///
170  SOPHUS_FUNC [[nodiscard]] Tangent log() const { return params(); }
171 
172  /// Returns 4x4 matrix representation of the instance.
173  ///
174  /// It has the following form:
175  ///
176  /// | I p |
177  /// | o 1 |
178  ///
179  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
180  Eigen::Matrix<Scalar, kMatrixDim, kMatrixDim> matrix;
181  matrix.setIdentity();
182  matrix.col(kM).template head<kM>() = params();
183  return matrix;
184  }

186  /// Group multiplication, are vector additions.
187  ///
188  template <class TOtherDerived>
189  SOPHUS_FUNC CartesianBase<TDerived, kM>& operator=(
190  CartesianBase<TOtherDerived, kM> const& other) {
191  params() = other.params();
192  return *this;
193  }
194 
195  /// Group multiplication, are vector additions.
196  ///
197  template <class TOtherDerived>
198  SOPHUS_FUNC CartesianSum<TOtherDerived> operator*(
199  CartesianBase<TOtherDerived, kM> const& other) const {
200  return CartesianSum<TOtherDerived>(params() + other.params());
201  }
202 
203  /// Group action on points, again just vector addition.
204  ///
205  template <
206  typename TPointDerived,
207  typename = typename std::enable_if<
208  IsFixedSizeVector<TPointDerived, kM>::value>::type>
209  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
210  Eigen::MatrixBase<TPointDerived> const& p) const {
211  return PointProduct<TPointDerived>(params() + p);
212  }
213 
214  /// Group action on homogeneous points. See above for more details.
215  ///
216  template <
217  typename THPointDerived,
218  typename = typename std::enable_if<
219  IsFixedSizeVector<THPointDerived, kMatrixDim>::value>::type>
220  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
221  Eigen::MatrixBase<THPointDerived> const& p) const {
222  auto const rp = *this * p.template head<kM>();
223  HomogeneousPointProduct<THPointDerived> r;
224  r << rp, p(kM);
225  return r;
226  }
227 
228  /// Group action on lines.
229  ///
230  SOPHUS_FUNC Line operator*(Line const& l) const {
231  return Line((*this) * l.origin(), l.direction());
232  }
233 
234  /// Group action on planes.
235  ///
236  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
237  return Hyperplane(p.normal(), p.offset() - params().dot(p.normal()));
238  }
239 
240  /// In-place group multiplication. This method is only valid if the return
241  /// type of the multiplication is compatible with this Cartesian's Scalar
242  /// type.
243  ///
244  template <
245  typename TOtherDerived,
246  typename = typename std::enable_if<
247  std::is_same<Scalar, ReturnScalar<TOtherDerived>>::value>::type>
248  SOPHUS_FUNC CartesianBase<TDerived, kM>& operator*=(
249  CartesianBase<TOtherDerived, kM> const& other) {
250  *static_cast<TDerived*>(this) = *this * other;
251  return *this;
252  }
253 
254  /// Mutator of params vector.
255  ///
256  SOPHUS_FUNC ParamsType& params() {
257  return static_cast<TDerived*>(this)->params();
258  }
259 
260  /// Accessor of params vector
261  ///
262  SOPHUS_FUNC [[nodiscard]] ParamsType const& params() const {
263  return static_cast<TDerived const*>(this)->params();
264  }

312  /// Accepts either M-vector or (M+1)x(M+1) matrices.
313  ///
314  template <class TD>
315  explicit SOPHUS_FUNC Cartesian(Eigen::MatrixBase<TD> const& m) {
316  static_assert(
317  std::is_same<typename Eigen::MatrixBase<TD>::Scalar, Scalar>::value,

324  } else {
325  SOPHUS_ASSERT(false, "{} {}", m.rows(), m.cols());
326  }

380  /// Returns the ith infinitesimal generators of Cartesian(M).
381  ///
382  /// The infinitesimal generators for e.g. the 3-dimensional case:
383  ///
384  /// ```
385  /// | 0 0 0 1 |
386  /// G_0 = | 0 0 0 0 |
387  /// | 0 0 0 0 |
388  /// | 0 0 0 0 |
389  ///
390  /// | 0 0 0 0 |
391  /// G_1 = | 0 0 0 1 |
392  /// | 0 0 0 0 |
393  /// | 0 0 0 0 |
394  ///
395  /// | 0 0 0 0 |
396  /// G_2 = | 0 0 0 0 |
397  /// | 0 0 0 1 |
398  /// | 0 0 0 0 |
399  /// ```
400  ///
401  /// Precondition: ``i`` must be in [0, M-1].
402  ///
403  SOPHUS_FUNC static Transformation generator(int i) {
404  SOPHUS_ASSERT(i >= 0 && i <= kM, "i should be in range [0,M-1].");
405  Tangent e;
406  e.setZero();
407  e[i] = Scalar(1);
408  return hat(e);
409  }
410 
411  /// Group exponential
412  ///
413  /// For Euclidean vector space, just the identity. Or to be more precise
414  /// it just constructs the (M+1xM+1) homogeneous matrix representation
415  // from the M-vector.
416  ///
417  SOPHUS_FUNC static Cartesian<Scalar, kM> exp(Tangent const& a) {
418  return Cartesian<Scalar, kM>(a);
419  }
420 
421  /// hat-operator
422  ///
423  /// Formally, the hat()-operator of Cartesian(M) is defined as
424  ///
425  /// ``hat(.): R^M -> R^{M+1xM+1}, hat(a) = sum_i a_i * G_i``
426  /// (for i=0,...,M-1)
427  ///
428  /// with ``G_i`` being the ith infinitesimal generator of Cartesian(M).
429  ///
430  /// The corresponding inverse is the vee()-operator, see below.
431  ///
432  SOPHUS_FUNC static Transformation hat(Tangent const& a) {
433  Transformation omega;
434  omega.setZero();
435  omega.col(kM).template head<kM>() = a.template head<kM>();
436  return omega;
437  }
438 
439  /// Lie bracket
440  ///
441  /// Always 0 for commutative groups.
442  SOPHUS_FUNC static Tangent lieBracket(
443  Tangent const& /*unused*/, Tangent const& /*unused*/) {
444  return Tangent::Zero();
445  }

447  /// Draws uniform samples in the range [-1, 1] per coordinates.
448  ///
449  template <class TUniformRandomBitGenerator>
450  static Cartesian sampleUniform(TUniformRandomBitGenerator& generator) {
451  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
452  Eigen::Vector<Scalar, kM> v;
453  for (int i = 0; i < kM; ++i) {
454  v[i] = uniform(generator);
455  }
456  return Cartesian(v);
457  }
458 
459  /// vee-operator
460  ///
461  /// This is the inverse of the hat()-operator, see above.
462  ///
463  SOPHUS_FUNC static Tangent vee(Transformation const& m) {
464  return m.col(kM).template head<kM>();
465  }

467  protected:

469 };
