17 /// Type-erased image with shared ownership, and read-only access to pixels.
18 /// Type is nullable.
19 template <
20  class TPredicate = AnyImagePredicate,
21  template <class> class TAllocator = Eigen::aligned_allocator>
22 class RuntimeImage : public RuntimeImageView<TPredicate> {
23  public:
24  /// Empty image.
25  RuntimeImage() = default;

27  /// Create type-erased image from Image.
28  ///
29  /// Ownership is shared between RuntimeImage and Image, and hence the
30  /// reference count will be increased by one (unless input is empty).
31  /// By design not "explicit".
32  template <class TPixel>
33  RuntimeImage(Image<TPixel, TAllocator> const& image)
34  : RuntimeImage(
35  image.shape(),
36  RuntimePixelType::fromTemplate<TPixel>(),
37  image.shared_) {
38  static_assert(TPredicate::template isTypeValid<TPixel>());
39  }

44  RuntimeImage(MutImage<TPixel>&& image)
45  : RuntimeImage(Image<TPixel>(std::move(image))) {
46  static_assert(TPredicate::template isTypeValid<TPixel>());
47  }

49  /// Create type-image image from provided shape and pixel type.
50  /// Pixel data is left uninitialized
51  RuntimeImage(ImageShape const& shape, RuntimePixelType const& pixel_type)
52  : RuntimeImage(
