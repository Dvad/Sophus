30  template <class TPixel>
31  static RuntimePixelType fromTemplate() {
32  return RuntimePixelType{
33  .number_type =
34  std::is_floating_point_v<typename ImageTraits<TPixel>::ChannelT>
35  ? NumberType::floating_point
36  : NumberType::fixed_point,
37  .num_channels = ImageTraits<TPixel>::kNumChannels,
38  .num_bytes_per_pixel_channel =

61 class RuntimeImageView {
62  public:
63  /// Create type-erased image view from ImageView.
64  ///
65  /// By design not "explicit".
66  template <class TPixel>
67  RuntimeImageView(ImageView<TPixel> const& image)
68  : RuntimeImageView(
69  image.shape(),
70  RuntimePixelType::fromTemplate<TPixel>(),
71  image.ptr()) {
72  static_assert(TPredicate::template isTypeValid<TPixel>());
73  }
74 
75  RuntimeImageView(
76  ImageShape const& image_shape,
77  RuntimePixelType const& pixel_type,
78  void const* ptr)
79  : shape_(image_shape),
80  pixel_type_(pixel_type),
81  ptr_(reinterpret_cast<uint8_t const*>(ptr)) {}
82 
83  /// Return true is this contains data of type TPixel.
84  template <class TPixel>
85  [[nodiscard]] bool has() const noexcept {
86  RuntimePixelType expected_type = RuntimePixelType::fromTemplate<TPixel>();
87  return expected_type == pixel_type_;
88  }
89 
90  /// Returns v-th row pointer.
91  ///
92  /// Precondition: v must be in [0, height).
93  [[nodiscard]] uint8_t const* rawRowPtr(int v) const {
94  return ((uint8_t*)(rawPtr()) + v * shape_.pitchBytes());
95  }
96 
97  [[nodiscard]] uint8_t const* rawPtr() const { return ptr_; }
98 
99  /// Number of bytes per channel of a single pixel.
100  ///
101  /// E.g. a pixel of Eigen::Matrix<uint8_t, 3, 1> has 1 byte per channel.
102  [[nodiscard]] int numBytesPerPixelChannel() const {
103  return pixel_type_.num_bytes_per_pixel_channel;
104  }
105  [[nodiscard]] NumberType numberType() const {
106  return pixel_type_.number_type;
107  }
108 
109  [[nodiscard]] ImageShape const& shape() const { return shape_; }
110 
111  [[nodiscard]] ImageSize const& imageSize() const {
112  return shape_.imageSize();
113  }
114 
115  [[nodiscard]] int width() const { return shape().width(); }
116  [[nodiscard]] int height() const { return shape().height(); }
117  [[nodiscard]] size_t pitchBytes() const { return shape().pitchBytes(); }
118  [[nodiscard]] size_t sizeBytes() const { return height() * pitchBytes(); }
119  [[nodiscard]] bool isEmpty() const { return this->rawPtr() == nullptr; }
120 
121  [[nodiscard]] RuntimePixelType pixelType() const { return pixel_type_; }
122  [[nodiscard]] int numChannels() const {
123  return this->pixel_type_.num_channels;
124  }
125 
126  /// Returns subview with shared ownership semantics of whole image.
127  [[nodiscard]] RuntimeImageView subview(
128  Eigen::Vector2i uv, sophus::ImageSize size) const {
129  SOPHUS_ASSERT(imageSize().contains(uv));
130  SOPHUS_ASSERT_LE(uv.x() + size.width, this->shape_.width());
131  SOPHUS_ASSERT_LE(uv.y() + size.height, this->shape_.height());
132 
133  auto const shape =
134  ImageShape::makeFromSizeAndPitchUnchecked(size, pitchBytes());
135  const size_t row_offset =
136  uv.x() * numBytesPerPixelChannel() * numChannels();
137  uint8_t const* ptr = this->rawPtr() + uv.y() * pitchBytes() + row_offset;
138  return RuntimeImageView{shape, this->pixel_type_, ptr};
139  }
140 
141  /// Returns typed image.
142  ///
143  /// Precondition: this->has<TPixel>()
144  template <class TPixel>
145  [[nodiscard]] ImageView<TPixel> imageView() const noexcept {
146  if (!this->has<TPixel>()) {
147  RuntimePixelType expected_type = RuntimePixelType::fromTemplate<TPixel>();
148 
149  SOPHUS_PANIC(
150  "expected type: {}\n"
151  "actual type: {}",
152  expected_type,
153  this->pixel_type_);
154  }
155 
156  return ImageView<TPixel>(
157  this->shape_, reinterpret_cast<TPixel const*>(ptr_));
158  }
159 
160  protected:
161  RuntimeImageView() = default;
162 
163  ImageShape shape_ = {}; // NOLINT
164  RuntimePixelType pixel_type_; // NOLINT
165  uint8_t const* ptr_; // NOLINT
166 };

168 /// Type-erased image with shared ownership, and read-only access to pixels.
169 /// Type is nullable.
170 template <
171  class TPredicate = AnyImagePredicate,
172  template <class> class TAllocator = Eigen::aligned_allocator>
173 class RuntimeImage : public RuntimeImageView<TPredicate> {
174  public:
175  /// Empty image.
176  RuntimeImage() = default;
177 
178  /// Create type-erased image from Image.
179  ///
180  /// Ownership is shared between RuntimeImage and Image, and hence the
181  /// reference count will be increased by one (unless input is empty).
182  /// By design not "explicit".
183  template <class TPixel>
184  RuntimeImage(Image<TPixel, TAllocator> const& image)
185  : RuntimeImage(
186  image.shape(),
187  RuntimePixelType::fromTemplate<TPixel>(),
188  image.shared_) {
189  static_assert(TPredicate::template isTypeValid<TPixel>());
190  }
191 
192  /// Create type-erased image from MutImage.
193  /// By design not "explicit".
194  template <class TPixel>
195  RuntimeImage(MutImage<TPixel>&& image)
196  : RuntimeImage(Image<TPixel>(std::move(image))) {
197  static_assert(TPredicate::template isTypeValid<TPixel>());
198  }
199 
200  /// Create type-image image from provided shape and pixel type.
201  /// Pixel data is left uninitialized
202  RuntimeImage(ImageShape const& shape, RuntimePixelType const& pixel_type)
203  : RuntimeImage(
204  shape,
205  pixel_type,
206  std::shared_ptr<uint8_t>(TAllocator<uint8_t>().allocate(
207  shape.height() * shape.pitchBytes()))) {
208  // TODO: Missing check on ImagePredicate against pixel_type
209  // has to be a runtime check, since we don't know at runtime.
210 
211  SOPHUS_ASSERT_LE(
212  shape.width() * pixel_type.num_channels *
213  pixel_type.num_bytes_per_pixel_channel,
214  (int)shape.pitchBytes());
215  }
216 
217  /// Create type-image image from provided size and pixel type.
218  /// Pixel data is left uninitialized
219  RuntimeImage(ImageSize const& size, RuntimePixelType const& pixel_type)
220  : RuntimeImage(
221  ImageShape::makeFromSizeAndPitch<uint8_t>(
222  size,
223  size.width * pixel_type.num_channels *
224  pixel_type.num_bytes_per_pixel_channel),
225  pixel_type) {}
226 
227  template <class TT>
228  static RuntimeImage makeCopyFrom(ImageView<TT> image_view) {
229  return Image<TT>::makeCopyFrom(image_view);
230  }
231 
232  /// Return true is this contains data of type TPixel.
233  template <class TPixel>
234  [[nodiscard]] bool has() const noexcept {
235  RuntimePixelType expected_type = RuntimePixelType::fromTemplate<TPixel>();
236  return expected_type == this->pixel_type_;
237  }
238 
239  /// Returns typed image.
240  ///
241  /// Precondition: this->has<TPixel>()
242  template <class TPixel>
243  [[nodiscard]] Image<TPixel, TAllocator> image() const noexcept {
244  if (!this->has<TPixel>()) {
245  RuntimePixelType expected_type = RuntimePixelType::fromTemplate<TPixel>();
246 
247  SOPHUS_PANIC(
248  "expected type: {}\n"
249  "actual type: {}",
250  expected_type,
251  this->pixel_type_);
252  }
253 
254  return Image<TPixel, TAllocator>(
255  ImageView<TPixel>(
256  this->shape_, reinterpret_cast<TPixel*>(shared_.get())),
257  shared_);
258  }
259 
260  template <class TPixel>
261  Image<TPixel, TAllocator> reinterpretAs(
262  ImageSize reinterpreted_size) const noexcept {
263  SOPHUS_ASSERT_LE(
264  reinterpreted_size.width * sizeof(TPixel), this->shape().pitch_bytes_);
265  SOPHUS_ASSERT_LE(reinterpreted_size.height, this->height());
266 
267  SOPHUS_UNIMPLEMENTED();
268  }
269 
270  [[nodiscard]] size_t useCount() const { return shared_.use_count(); }
271 
272  private:
273  // Private constructor mainly available for constructing sub-views
274  RuntimeImage(
275  ImageShape shape,
276  RuntimePixelType pixel_type,
277  std::shared_ptr<uint8_t> shared)
278  : RuntimeImageView<TPredicate>(shape, pixel_type, shared.get()),
279  shared_(shared) {}
280 
281  std::shared_ptr<uint8_t> shared_;
282 };

318 namespace detail {
319 // Call UserFunc with TRuntimeImage cast to the appropriate concrete type
320 // from the options in PixelTypes...
321 template <class TUserFunc, typename TRuntimeImage, typename... TPixelTypes>
322 struct VisitImpl;
323 
324 // base case
325 template <class TUserFunc, typename TRuntimeImage, typename TPixelType>
326 struct VisitImpl<TUserFunc, TRuntimeImage, std::variant<TPixelType>> {
327  static void visit(TUserFunc&& func, TRuntimeImage const& image) {
328  if (image.pixelType().template is<TPixelType>()) {
329  func(image.template imageView<TPixelType>());
330  }
331  }
332 };
333 
334 // inductive case
335 template <
336  typename TUserFunc,
337  typename TRuntimeImage,
338  typename TPixelType,
339  typename... TRest>
340 struct VisitImpl<TUserFunc, TRuntimeImage, std::variant<TPixelType, TRest...>> {
341  static void visit(TUserFunc&& func, TRuntimeImage const& image) {
342  if (image.pixelType().template is<TPixelType>()) {
343  func(image.template imageView<TPixelType>());
344  } else {
345  VisitImpl<TUserFunc, TRuntimeImage, std::variant<TRest...>>::visit(
346  std::forward<TUserFunc>(func), image);
347  }
348  }
349 };
350 } // namespace detail
