141 /// If it is false that `lhs` is near `rhs` according to threshold `thr`, print
142 /// formatted error message and then panic.
143 ///
144 /// `lhs` and `rhs` are near, if maxMetric(lhs, rhs) < thr.
145 #define FARM_ASSERT_NEAR(lhs, rhs, thr, ...) \
146  do { \
147  auto nrm = farm_ng::maxMetric((lhs), (rhs)); \
148  if (!(nrm < (thr))) { \
149  FARM_IMPL_LOG_HEADER("PANIC: ASSERT_NEAR failed"); \
150  FARM_IMPL_LOG_PRINTLN( \
151  "Not true: {} near {}; has error of {} (thr: " \
152  "{})\n{}\n{}\nvs.\n{}\n{}", \
153  #lhs, \
154  #rhs, \
155  nrm, \
156  thr, \
157  #lhs, \
158  lhs, \
159  #rhs, \
160  rhs); \
161  FARM_IMPL_LOG_PRINTLN(__VA_ARGS__); \
162  FARM_IMPL_ABORT(); \
163  } \
164  } while (false)

176 template <class TContainer>
177 auto checkedAtContiguousContainer(
178  TContainer& container,
179  size_t index,
180  char const* container_cstr,
181  char const* index_cstr,
182  std::string const& file,
183  int line,
184  std::string const& str) -> decltype(container[index]) {
185  if (index >= container.size()) {
186  FARM_IMPL_LOG_PRINTLN("[FARM_AT in {}:{}]", file, line);
187  FARM_IMPL_LOG_PRINTLN(
188  "index `{}` (={}) not in contiguous container `{}` of size `{}`",
189  index_cstr,
190  index,
191  container_cstr,
192  container.size());
193  if (!str.empty()) {
194  ::fmt::print(stderr, "{}", str);
195  }
196  FARM_IMPL_ABORT();
197  }
198  return container[index];
199 }
